16pub struct InstanceState {
17    allowed_databases: Arc<HashSet<String>>,
18    /// A resource table of connections.
19    connections: spin_resource_table::Table<Box<dyn Connection>>,
20    /// A map from database label to connection creators.
21    connection_creators: HashMap<String, Arc<dyn ConnectionCreator>>,
22    otel: OtelFactorState,
23}
24
25impl InstanceState {
26    /// Create a new `InstanceState`
27    ///
28    /// Takes the list of allowed databases, and a function for getting a connection creator given a database label.
29    pub fn new(
30        allowed_databases: Arc<HashSet<String>>,
31        connection_creators: HashMap<String, Arc<dyn ConnectionCreator>>,
32        otel: OtelFactorState,
33    ) -> Self {
34        Self {
35            allowed_databases,
36            connections: spin_resource_table::Table::new(256),
37            connection_creators,
38            otel,
39        }
40    }
41
42    /// Get a connection for a given database label.
43    fn get_connection<T: 'static>(
44        &self,
45        connection: Resource<T>,
46    ) -> Result<&dyn Connection, v3::Error> {
47        self.connections
48            .get(connection.rep())
49            .map(|conn| conn.as_ref())
50            .ok_or(v3::Error::InvalidConnection)
51    }
52
53    async fn open_impl<T: 'static>(&mut self, database: String) -> Result<Resource<T>, v3::Error> {
54        if !self.allowed_databases.contains(&database) {
55            return Err(v3::Error::AccessDenied);
56        }
57        let conn = self
58            .connection_creators
59            .get(&database)
60            .ok_or(v3::Error::NoSuchDatabase)?
61            .create_connection(&database)
62            .await?;
63        tracing::Span::current().record(
64            "sqlite.backend",
65            conn.summary().as_deref().unwrap_or("unknown"),
66        );
67        self.connections
68            .push(conn)
69            .map_err(|()| v3::Error::Io("too many connections opened".to_string()))
70            .map(Resource::new_own)
71    }
72
73    async fn execute_impl<T: 'static>(
74        &mut self,
75        connection: Resource<T>,
76        query: String,
77        parameters: Vec<v3::Value>,
78    ) -> Result<v3::QueryResult, v3::Error> {
79        let conn = self.get_connection(connection)?;
80        tracing::Span::current().record(
81            "sqlite.backend",
82            conn.summary().as_deref().unwrap_or("unknown"),
83        );
84        conn.query(&query, parameters, MAX_HOST_BUFFERED_BYTES)
85            .await
86    }
87
88    /// Get the set of allowed databases.
89    pub fn allowed_databases(&self) -> &HashSet<String> {
90        &self.allowed_databases
91    }
92}

102impl v3::HostConnection for InstanceState {
103    #[instrument(name = "spin_sqlite.open", skip(self), err(level = Level::INFO), fields(otel.kind = "client", db.system = "sqlite", sqlite.backend = Empty))]

105        self.open_impl(database).await
106    }

108    #[instrument(name = "spin_sqlite.execute", skip(self, connection, parameters), err(level = Level::INFO), fields(otel.kind = "client", db.system = "sqlite", otel.name = query, sqlite.backend = Empty))]
109    async fn execute(
110        &mut self,
111        connection: Resource<v3::Connection>,
112        query: String,
113        parameters: Vec<v3::Value>,
114    ) -> Result<v3::QueryResult, v3::Error> {
115        self.execute_impl(connection, query, parameters).await
116    }
117
118    async fn changes(
119        &mut self,
120        connection: Resource<v3::Connection>,
121    ) -> spin_factors::wasmtime::Result<u64> {
122        let conn = match self.get_connection(connection) {
123            Ok(c) => c,
124            Err(err) => return Err(err.into()),
125        };
126        tracing::Span::current().record(
127            "sqlite.backend",
128            conn.summary().as_deref().unwrap_or("unknown"),
129        );
130        conn.changes().await.map_err(|e| e.into())
131    }
132
133    async fn last_insert_rowid(
134        &mut self,
135        connection: Resource<v3::Connection>,
136    ) -> spin_factors::wasmtime::Result<i64> {
137        let conn = match self.get_connection(connection) {
138            Ok(c) => c,
139            Err(err) => return Err(err.into()),
140        };
141        tracing::Span::current().record(
142            "sqlite.backend",
143            conn.summary().as_deref().unwrap_or("unknown"),
144        );

146    }

160impl v2::HostConnection for InstanceState {
161    #[instrument(name = "spin_sqlite.open", skip(self), err(level = Level::INFO), fields(otel.kind = "client", db.system = "sqlite", sqlite.backend = Empty))]
162    async fn open(&mut self, database: String) -> Result<Resource<v2::Connection>, v2::Error> {
163        self.otel.reparent_tracing_span();
164        self.open_impl(database).await.map_err(to_v2_error)
165    }
166
167    #[instrument(name = "spin_sqlite.execute", skip(self, connection, parameters), err(level = Level::INFO), fields(otel.kind = "client", db.system = "sqlite", otel.name = query, sqlite.backend = Empty))]
168    async fn execute(
169        &mut self,
170        connection: Resource<v2::Connection>,
171        query: String,
172        parameters: Vec<v2::Value>,
173    ) -> Result<v2::QueryResult, v2::Error> {
174        self.otel.reparent_tracing_span();
175        self.execute_impl(
176            connection,
177            query,
178            parameters.into_iter().map(from_v2_value).collect(),
179        )
180        .await
181        .map(to_v2_query_result)
182        .map_err(to_v2_error)
183    }
184
185    async fn drop(&mut self, connection: Resource<v2::Connection>) -> anyhow::Result<()> {
186        let _ = self.connections.remove(connection.rep());
187        Ok(())
188    }
189}

191impl v1::Host for InstanceState {
192    async fn open(&mut self, database: String) -> Result<u32, v1::Error> {
193        let result = <Self as v3::HostConnection>::open(self, database).await;
194        result.map_err(to_legacy_error).map(|s| s.rep())
195    }
196
197    async fn execute(
198        &mut self,
199        connection: u32,
200        query: String,
201        parameters: Vec<spin_world::v1::sqlite::Value>,
202    ) -> Result<spin_world::v1::sqlite::QueryResult, v1::Error> {
203        let this = Resource::new_borrow(connection);
204        let result = <Self as v3::HostConnection>::execute(
205            self,
206            this,
207            query,
208            parameters.into_iter().map(from_legacy_value).collect(),
209        )
210        .await;
211        result.map_err(to_legacy_error).map(to_legacy_query_result)
212    }
213
214    async fn close(&mut self, connection: u32) -> anyhow::Result<()> {
215        <Self as v2::HostConnection>::drop(self, Resource::new_own(connection)).await
216    }

223fn to_v2_error(error: v3::Error) -> v2::Error {
224    match error {
225        v3::Error::NoSuchDatabase => v2::Error::NoSuchDatabase,
226        v3::Error::AccessDenied => v2::Error::AccessDenied,
227        v3::Error::InvalidConnection => v2::Error::InvalidConnection,
228        v3::Error::DatabaseFull => v2::Error::DatabaseFull,
229        v3::Error::Io(s) => v2::Error::Io(s),
230    }
231}
232
233fn to_legacy_error(error: v3::Error) -> v1::Error {
234    match error {
235        v3::Error::NoSuchDatabase => v1::Error::NoSuchDatabase,
236        v3::Error::AccessDenied => v1::Error::AccessDenied,
237        v3::Error::InvalidConnection => v1::Error::InvalidConnection,
238        v3::Error::DatabaseFull => v1::Error::DatabaseFull,
239        v3::Error::Io(s) => v1::Error::Io(s),
240    }
241}
