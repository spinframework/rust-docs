80impl From<CreateChatCompletionResponse> for wasi_llm::InferencingResult {
81    fn from(value: CreateChatCompletionResponse) -> Self {
82        Self {
83            text: value
84                .choices
85                .first()
86                .map_or_else(String::new, |c| c.message.content.clone()),
87            usage: wasi_llm::InferencingUsage {
88                prompt_token_count: value.usage.prompt_tokens,
89                generated_token_count: value.usage.completion_tokens,

118#[derive(Serialize, Debug)]
119pub(crate) enum Role {
120    #[serde(rename = "system")]
121    System,
122    #[serde(rename = "user")]
123    User,
124    #[serde(rename = "assistant")]
125    Assistant,
126    #[serde(rename = "tool")]
127    Tool,
128}
129
130impl TryFrom<&str> for Role {
131    type Error = wasi_llm::Error;
132
133    fn try_from(value: &str) -> Result<Self, Self::Error> {
134        match value {
135            "system" => Ok(Role::System),
136            "user" => Ok(Role::User),
137            "assistant" => Ok(Role::Assistant),
138            "tool" => Ok(Role::Tool),
139            _ => Err(wasi_llm::Error::InvalidInput(format!(
140                "{value} not a valid role"
141            ))),
142        }
143    }
144}
