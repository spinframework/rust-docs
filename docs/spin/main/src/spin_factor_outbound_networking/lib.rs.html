45impl Factor for OutboundNetworkingFactor {
46    type RuntimeConfig = RuntimeConfig;
47    type AppState = AppState;
48    type InstanceBuilder = InstanceBuilder;
49
50    fn configure_app<T: RuntimeFactors>(
51        &self,
52        mut ctx: ConfigureAppContext<T, Self>,
53    ) -> anyhow::Result<Self::AppState> {
54        // Extract allowed_outbound_hosts for all components
55        let component_allowed_hosts = ctx
56            .app()
57            .components()
58            .map(|component| {
59                Ok((
60                    component.id().to_string(),
61                    allowed_outbound_hosts(&component)?
62                        .into_boxed_slice()
63                        .into(),
64                ))
65            })
66            .collect::<anyhow::Result<_>>()?;
67
68        let runtime_config = match ctx.take_runtime_config() {
69            Some(cfg) => cfg,
70            // The default RuntimeConfig provides default TLS client configs
71            None => RuntimeConfig::new([])?,
72        };
73
74        Ok(AppState {
75            component_allowed_hosts,
76            runtime_config,
77        })
78    }
79
80    fn prepare<T: RuntimeFactors>(
81        &self,
82        mut ctx: PrepareContext<T, Self>,
83    ) -> anyhow::Result<Self::InstanceBuilder> {
84        let hosts = ctx
85            .app_state()
86            .component_allowed_hosts
87            .get(ctx.app_component().id())
88            .cloned()
89            .context("missing component allowed hosts")?;
90        let resolver = ctx
91            .instance_builder::<VariablesFactor>()?
92            .expression_resolver()
93            .clone();
94        let allowed_hosts_future = async move {
95            let prepared = resolver.prepare().await?;
96            AllowedHostsConfig::parse(&hosts, &prepared)
97        }
98        .map(|res| res.map(Arc::new).map_err(Arc::new))
99        .boxed()
100        .shared();
101
102        match ctx.instance_builder::<WasiFactor>() {
103            Ok(wasi_builder) => {
104                // Update Wasi socket allowed ports
105                let allowed_hosts = OutboundAllowedHosts {
106                    allowed_hosts_future: allowed_hosts_future.clone(),
107                    disallowed_host_handler: self.disallowed_host_handler.clone(),
108                };
109                wasi_builder.outbound_socket_addr_check(move |addr, addr_use| {
110                    let allowed_hosts = allowed_hosts.clone();
111                    async move {
112                        let scheme = match addr_use {
113                            SocketAddrUse::TcpBind => return false,
114                            SocketAddrUse::TcpConnect => "tcp",
115                            SocketAddrUse::UdpBind
116                            | SocketAddrUse::UdpConnect
117                            | SocketAddrUse::UdpOutgoingDatagram => "udp",
118                        };
119                        allowed_hosts
120                            .check_url(&addr.to_string(), scheme)
121                            .await
122                            .unwrap_or(
123                                // TODO: should this trap (somehow)?
124                                false,
125                            )
126                    }
127                });
128            }
129            Err(Error::NoSuchFactor(_)) => (), // no WasiFactor to configure; that's OK
130            Err(err) => return Err(err.into()),
131        }
132
133        let component_tls_configs = ctx
134            .app_state()
135            .runtime_config
136            .get_component_tls_configs(ctx.app_component().id());
137
138        Ok(InstanceBuilder {
139            allowed_hosts_future,
140            component_tls_configs,
141            disallowed_host_handler: self.disallowed_host_handler.clone(),
142        })
143    }
144}

178/// A check for whether a URL is allowed by the outbound networking configuration.
179#[derive(Clone)]
180pub struct OutboundAllowedHosts {
181    allowed_hosts_future: SharedFutureResult<AllowedHostsConfig>,
182    disallowed_host_handler: Option<Arc<dyn DisallowedHostHandler>>,
183}
184
185impl OutboundAllowedHosts {
186    /// Checks address against allowed hosts
187    ///
188    /// Calls the [`DisallowedHostHandler`] if set and URL is disallowed.
189    /// If `url` cannot be parsed, `{scheme}://` is prepended to `url` and retried.
190    pub async fn check_url(&self, url: &str, scheme: &str) -> anyhow::Result<bool> {
191        tracing::debug!("Checking outbound networking request to '{url}'");
192        let url = match OutboundUrl::parse(url, scheme) {
193            Ok(url) => url,
194            Err(err) => {
195                tracing::warn!(%err,
196                    "A component tried to make a request to a url that could not be parsed: {url}",
197                );
198                return Ok(false);
199            }
200        };
201
202        let allowed_hosts = self.resolve().await?;
203        let is_allowed = allowed_hosts.allows(&url);
204        if !is_allowed {
205            tracing::debug!("Disallowed outbound networking request to '{url}'");
206            self.report_disallowed_host(url.scheme(), &url.authority());
207        }
208        Ok(is_allowed)
209    }
210
211    /// Checks if allowed hosts permit relative requests
212    ///
213    /// Calls the [`DisallowedHostHandler`] if set and relative requests are
214    /// disallowed.
215    pub async fn check_relative_url(&self, schemes: &[&str]) -> anyhow::Result<bool> {
216        tracing::debug!("Checking relative outbound networking request with schemes {schemes:?}");
217        let allowed_hosts = self.resolve().await?;
218        let is_allowed = allowed_hosts.allows_relative_url(schemes);
219        if !is_allowed {
220            tracing::debug!(
221                "Disallowed relative outbound networking request with schemes {schemes:?}"
222            );
223            let scheme = schemes.first().unwrap_or(&"");
224            self.report_disallowed_host(scheme, "self");
225        }
226        Ok(is_allowed)
227    }
228
229    async fn resolve(&self) -> anyhow::Result<Arc<AllowedHostsConfig>> {
230        self.allowed_hosts_future.clone().await.map_err(|err| {
231            tracing::error!(%err, "Error resolving variables when checking request against allowed outbound hosts");
232            anyhow::Error::msg(err)
233        })
234    }
235
236    fn report_disallowed_host(&self, scheme: &str, authority: &str) {
237        if let Some(handler) = &self.disallowed_host_handler {
238            handler.handle_disallowed_host(scheme, authority);
239        }
240    }
241}

243pub trait DisallowedHostHandler: Send + Sync {
244    fn handle_disallowed_host(&self, scheme: &str, authority: &str);
245}
246
247impl<F: Fn(&str, &str) + Send + Sync> DisallowedHostHandler for F {
248    fn handle_disallowed_host(&self, scheme: &str, authority: &str) {
249        self(scheme, authority);
250    }

253/// Records the address host, port, and database as fields on the current tracing span.
254///
255/// This should only be called from within a function that has been instrumented with a span.
256///
257/// The following fields must be pre-declared as empty on the span or they will not show up.
258/// ```
259/// use tracing::field::Empty;
260/// #[tracing::instrument(fields(db.address = Empty, server.port = Empty, db.namespace = Empty))]
261/// fn open() {}
262/// ```
263pub fn record_address_fields(address: &str) {
264    if let Ok(url) = Url::parse(address) {
265        let span = tracing::Span::current();
266        span.record("db.address", url.host_str().unwrap_or_default());
267        span.record("server.port", url.port().unwrap_or_default());
268        span.record("db.namespace", url.path().trim_start_matches('/'));
269    }
