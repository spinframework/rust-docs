6const ALLOWED_HOSTS_KEY: MetadataKey<Vec<String>> = MetadataKey::new("allowed_outbound_hosts");
7const ALLOWED_HTTP_KEY: MetadataKey<Vec<String>> = MetadataKey::new("allowed_http_hosts");
8
9/// Get the raw values of the `allowed_outbound_hosts` locked app metadata key.
10///
11/// This has support for converting the old `allowed_http_hosts` key to the new `allowed_outbound_hosts` key.
12pub fn allowed_outbound_hosts(component: &AppComponent) -> anyhow::Result<Vec<String>> {
13    let mut allowed_hosts = component
14        .get_metadata(ALLOWED_HOSTS_KEY)
15        .with_context(|| {
16            format!(
17                "locked app metadata was malformed for key {}",
18                ALLOWED_HOSTS_KEY.as_ref()
19            )
20        })?
21        .unwrap_or_default();
22    let allowed_http = component
23        .get_metadata(ALLOWED_HTTP_KEY)
24        .map(|h| h.unwrap_or_default())
25        .unwrap_or_default();
26    let converted =
27        spin_manifest::compat::convert_allowed_http_to_allowed_hosts(&allowed_http, false)
28            .unwrap_or_default();
29    allowed_hosts.extend(converted);
30    Ok(allowed_hosts)
31}

33/// Validates that all service chaining of an app will be satisfied by the
34/// supplied subset of components.
35///
36/// This does a best effort look up of components that are
37/// allowed to be accessed through service chaining and will error early if a
38/// component is configured to to chain to another component that is not
39/// retained. All wildcard service chaining is disallowed and all templated URLs
40/// are ignored.
41pub fn validate_service_chaining_for_components(
42    app: &App,
43    retained_components: &[&str],
44) -> anyhow::Result<()> {
45    app
46        .triggers().try_for_each(|t| {
47            let Ok(component) = t.component() else  { return Ok(()) };
48            if retained_components.contains(&component.id()) {
49            let allowed_hosts = allowed_outbound_hosts(&component).context("failed to get allowed hosts")?;
50            for host in allowed_hosts {
51                // Templated URLs are not yet resolved at this point, so ignore unresolvable URIs
52                if let Ok(uri) = host.parse::<http::Uri>() {
53                    if let Some(chaining_target) = parse_service_chaining_target(&uri) {
54                        if !retained_components.contains(&chaining_target.as_ref()) {
55                            if chaining_target == "*" {
56                                return  Err(anyhow::anyhow!("Selected component '{}' cannot use wildcard service chaining: allowed_outbound_hosts = [\"http://*.spin.internal\"]", component.id()));
57                            }
58                            return  Err(anyhow::anyhow!(
59                                "Selected component '{}' cannot use service chaining to unselected component: allowed_outbound_hosts = [\"http://{}.spin.internal\"]",
60                                component.id(), chaining_target
61                            ));
62                        }
63                    }
64                }
65            }
66        }
67        anyhow::Ok(())
68    })?;
69
70    Ok(())
71}

77    #[tokio::test]
78    async fn validate_service_chaining_for_components_fails() {
79        let manifest = toml::toml! {
80            spin_manifest_version = 2
81
82            [application]
83            name = "test-app"
84
85            [[trigger.test-trigger]]
86            component = "empty"
87
88            [component.empty]
89            source = "does-not-exist.wasm"
90            allowed_outbound_hosts = ["http://another.spin.internal"]
91
92            [[trigger.another-trigger]]
93            component = "another"
94
95            [component.another]
96            source = "does-not-exist.wasm"
97
98            [[trigger.third-trigger]]
99            component = "third"
100
101            [component.third]
102            source = "does-not-exist.wasm"
103            allowed_outbound_hosts = ["http://*.spin.internal"]
104        };
105        let locked_app = spin_factors_test::build_locked_app(&manifest)
106            .await
107            .expect("could not build locked app");
108        let app = App::new("unused", locked_app);
109        let Err(e) = validate_service_chaining_for_components(&app, &["empty"]) else {
110            panic!("Expected service chaining to non-retained component error");
111        };
112        assert_eq!(
113            e.to_string(),
114            "Selected component 'empty' cannot use service chaining to unselected component: allowed_outbound_hosts = [\"http://another.spin.internal\"]"
115        );
116        let Err(e) = validate_service_chaining_for_components(&app, &["third", "another"]) else {
117            panic!("Expected wildcard service chaining error");
118        };
119        assert_eq!(
120            e.to_string(),
121            "Selected component 'third' cannot use wildcard service chaining: allowed_outbound_hosts = [\"http://*.spin.internal\"]"
122        );
123        assert!(validate_service_chaining_for_components(&app, &["another"]).is_ok());
124    }
