10pub async fn extract_wits_into(
11    source: impl ExactSizeIterator<Item = (&DependencyName, &ComponentDependency)>,
12    app_root: impl AsRef<Path>,
13    dest_file: impl AsRef<Path>,
14) -> anyhow::Result<()> {
15    if source.len() == 0 {
16        if dest_file.as_ref().is_file() {
17            _ = tokio::fs::remove_file(dest_file).await; // Not an error if tidying fails
18        }
19        return Ok(());
20    }
21
22    let wit_text = extract_wits(source, app_root).await?;
23
24    tokio::fs::create_dir_all(
25        dest_file
26            .as_ref()
27            .parent()
28            .context("file is root directory")?,
29    )
30    .await?;
31    tokio::fs::write(dest_file, wit_text.as_bytes()).await?;
32
33    Ok(())
34}

36pub async fn extract_wits(
37    source: impl Iterator<Item = (&DependencyName, &ComponentDependency)>,
38    app_root: impl AsRef<Path>,
39) -> anyhow::Result<String> {
40    let loader = WasmLoader::new(app_root.as_ref().to_owned(), None, None).await?;
41
42    let mut package_wits = indexmap::IndexMap::new();
43
44    let mut aggregating_resolve = wit_parser::Resolve::default();
45    let aggregating_pkg_id =
46        aggregating_resolve.push_str("dummy.wit", "package root:component;\n\nworld root {}")?;
47    let aggregating_world_id =
48        aggregating_resolve.select_world(&[aggregating_pkg_id], Some("root"))?;
49
50    // TODO: figure out what to do if we import two itfs from same dep
51    for (index, (dependency_name, dependency)) in source.enumerate() {
52        let import_name = match dependency_name {
53            DependencyName::Plain(_) => None,
54            DependencyName::Package(dependency_package_name) => {
55                dependency_package_name.interface.as_ref()
56            }
57        };
58
59        let (wasm_path, export) = loader
60            .load_component_dependency(dependency_name, dependency)
61            .await
62            .with_context(|| format!("failed to load dependency {dependency_name}"))?;
63        let wasm_bytes = tokio::fs::read(&wasm_path).await?;
64
65        let decoded = read_wasm(&wasm_bytes)?;
66        let decoded = match export {
67            None => decoded,
68            Some(export) => {
69                munge_aliased_export(decoded, &export, dependency_name).with_context(|| {
70                    format!("failed to map named export {export} to dependency {dependency_name}")
71                })?
72            }
73        };
74        let impo_world = format!("impo-world{index}");
75        let importised = importize(decoded, Some(&impo_world))
76            .with_context(|| format!("failed to map importize dependency {dependency_name}"))?;
77
78        let imports = match import_name {
79            None => all_imports(&importised),
80            Some(itf) => one_import(&importised, itf.as_ref())?,
81        };
82
83        // Capture WITs for all packages used in the importised thing.
84        // Things like WASI packages may be depended on by multiple packages
85        // so we index on the package name to avoid emitting them twice.
86
87        let root_pkg = importised.package();
88        let useful_pkgs = importised
89            .resolve()
90            .packages
91            .iter()
92            .map(|p| p.0)
93            .filter(|pid| *pid != root_pkg)
94            .collect::<Vec<_>>();
95
96        for p in &useful_pkgs {
97            let pkg_name = importised
98                .resolve()
99                .packages
100                .get(*p)
101                .context("package not found in importised (id lookup failed)")? // shouldn't happen
102                .name
103                .clone();
104            let output = wit_component::OutputToString::default();
105            let mut printer = wit_component::WitPrinter::new(output);
106            printer.print_package(importised.resolve(), *p, false)?;
107            package_wits.insert(pkg_name, printer.output.to_string());
108        }
109
110        // Now add the imports to the aggregating component import world
111
112        let remap = aggregating_resolve.merge(importised.resolve().clone())?;
113        for iid in imports {
114            let mapped_iid = remap.map_interface(iid, Span::default())?;
115            let wk = wit_parser::WorldKey::Interface(mapped_iid);
116            let world_item = wit_parser::WorldItem::Interface {
117                id: mapped_iid,
118                stability: wit_parser::Stability::Unknown,
119                span: Span::default(),
120            };
121            let previous_world_item = aggregating_resolve
122                .worlds
123                .get_mut(aggregating_world_id)
124                .context("aggregated dependency world doesn't exist")? // shouldn't happen
125                .imports
126                .insert(wk, world_item);
127            if let Some(previous_world_item) = previous_world_item {
128                debug_assert!({
129                    let wit_parser::WorldItem::Interface { id, .. } = previous_world_item else {
130                        debug_assert!(
131                            false,
132                            "previous WorldItem did not match inserted WorldItem: not an interface"
133                        );
134                        break;
135                    };
136                    debug_assert_eq!(
137                        id, mapped_iid,
138                        "replacing WorldItem had a different InterfaceId from replaced one"
139                    );
140
141                    true
142                });
143            }
144        }
145    }
146
147    // Text for the root package and world(s)
148    let world_output = wit_component::OutputToString::default();
149    let mut world_printer = wit_component::WitPrinter::new(world_output);
150    world_printer.print(&aggregating_resolve, aggregating_pkg_id, &[])?;
151
152    let mut buf = String::new();
153
154    // Print the root package and the world(s) with the imports
155    buf.push_str(&world_printer.output.to_string());
156
157    // Print each package
158    for package_wit in package_wits.values() {
159        buf.push_str(package_wit);
160    }
161
162    Ok(buf)
163}

165fn munge_aliased_export(
166    decoded: DecodedWasm,
167    export: &str,
168    new_name: &DependencyName,
169) -> anyhow::Result<DecodedWasm> {
170    let export_qname = spin_serde::DependencyPackageName::try_from(export.to_string())?;
171    let Some(export_itf_name) = export_qname.interface.as_ref() else {
172        anyhow::bail!("the export name should be a qualified interface name - {export_qname} doesn't specify interface");
173    };
174    let export_pkg_name = wit_parser::PackageName {
175        namespace: export_qname.package.namespace().to_string(),
176        name: export_qname.package.name().to_string(),
177        version: export_qname.version,
178    };
179
180    let DependencyName::Package(new_name) = new_name else {
181        anyhow::bail!(
182            "the dependency name should be a qualified interface name - {new_name} not qualified"
183        );
184    };
185    let Some(new_itf_name) = new_name.interface.as_ref() else {
186        anyhow::bail!(
187            "the dependency name should be a qualified interface name - {new_name} doesn't specify an interface"
188        );
189    };
190    let new_pkg_name = wit_parser::PackageName {
191        namespace: new_name.package.namespace().to_string(),
192        name: new_name.package.name().to_string(),
193        version: new_name.version.clone(),
194    };
195
196    let (mut resolve, decode_id) = match decoded {
197        DecodedWasm::WitPackage(resolve, id) => (resolve, WorldOrPackageId::Package(id)),
198        DecodedWasm::Component(resolve, id) => (resolve, WorldOrPackageId::World(id)),
199    };
200
201    // Two scenarios:
202    // 1. The new name is in a package that is already in the Resolve
203    //    1a. The package already contains an interface with the right name
204    //    1b. The package does not already contain an interface with the right name
205    // 2. The new name is in a package that is NOT already in the Resolve
206
207    let existing_pkg = resolve
208        .packages
209        .iter()
210        .find(|(_pkg_id, pkg)| pkg.name == new_pkg_name);
211
212    // We address the first level by creating the new-name package if it doesn't exist
213    let (inserting_into_pkg_id, inserting_into_pkg) = match existing_pkg {
214        Some(tuple) => tuple,
215        None => {
216            // insert the needed package
217            let package_wit = format!("package {new_pkg_name};");
218            let pkg_id = resolve
219                .push_str(
220                    std::env::current_dir().context("no current dir")?, // unused
221                    &package_wit,
222                )
223                .with_context(|| format!("failed to create import alias package {new_pkg_name}"))?;
224            let pkg = resolve
225                .packages
226                .get(pkg_id)
227                .context("export alias package created but doesn't exist")?; // shouldn't happen
228            (pkg_id, pkg)
229        }
230    };
231
232    // Second level asks if the new-name package already contains the interface
233    let existing_itf = inserting_into_pkg.interfaces.get(new_itf_name.as_ref());
234    if existing_itf.is_some() {
235        // This makes the questionable assumption that the matchingly interface already
236        // in the package is the same as the export. E.g. given "a:b/i" = { export = "c:d/i" }
237        // where the dep contains an `a:b` package with an interface named `i`, we will generate
238        // from that rather than emitting the `c:d/i` WIT for `a:b/i`. We could be smarter about
239        // this but we don't know if other things depend on `a:b/i` so replacing it could result
240        // in a bad WIT. If this becomes a problem we could maybe try emitting it as something
241        // like `a:b/i-from-c-d` but I'd prefer to cross that bridge when we come to it.
242        return Ok(decode_id.make_decoded_wasm(resolve));
243    }
244
245    // The new-name package does not contain the interface: we need to slurp the EXPORTED itf into the `inserting_into`
246    // package under the NEW (importing) interface name
247    let Some(export_pkg_id) = resolve.package_names.get(&export_pkg_name) else {
248        anyhow::bail!(
249            "export is from a package ({}) that doesn't exist (name lookup failed)",
250            export_pkg_name
251        );
252    };
253    let Some(export_pkg) = resolve.packages.get(*export_pkg_id) else {
254        anyhow::bail!(
255            "export is from a package ({}) that doesn't exist (id lookup failed)",
256            export_pkg_name
257        );
258    };
259    let Some(export_itf_id) = export_pkg.interfaces.get(export_itf_name.as_ref()) else {
260        anyhow::bail!(
261            "export package ({}) doesn't contain interface {} (name lookup failed)",
262            export_pkg_name,
263            export_itf_name
264        );
265    };
266    let Some(export_itf) = resolve.interfaces.get(*export_itf_id) else {
267        anyhow::bail!(
268            "export package ({}) doesn't contain interface {} (id lookup failed)",
269            export_pkg_name,
270            export_itf_name
271        );
272    };
273
274    // Create the new-name interface by cloning the export interface
275    let mut export_itf = export_itf.clone();
276    export_itf.package = Some(inserting_into_pkg_id);
277    export_itf.name = Some(new_itf_name.to_string());
278
279    // Add the new-name interface to the resolve and to the new-name package
280    let export_itf_id_new = resolve.interfaces.alloc(export_itf);
281    let inserting_into_pkg_mut = resolve
282        .packages
283        .get_mut(inserting_into_pkg_id)
284        .context("package id lookup that succeeded before failed now")?; // we re-lookup to get around a "mutable borrow at the same time as immutable borrow" woe
285    inserting_into_pkg_mut
286        .interfaces
287        .insert(new_itf_name.to_string(), export_itf_id_new);
288
289    let decoded = decode_id.make_decoded_wasm(resolve);
290
291    Ok(decoded)
292}

294enum WorldOrPackageId {
295    Package(wit_parser::PackageId),
296    World(wit_parser::WorldId),
297}
298
299impl WorldOrPackageId {
300    pub fn make_decoded_wasm(&self, resolve: wit_parser::Resolve) -> DecodedWasm {
301        match self {
302            Self::Package(id) => DecodedWasm::WitPackage(resolve, *id),
303            Self::World(id) => DecodedWasm::Component(resolve, *id),
304        }
305    }
306}

308fn all_imports(wasm: &DecodedWasm) -> Vec<wit_parser::InterfaceId> {
309    wasm.resolve()
310        .worlds
311        .iter()
312        .flat_map(|(_wid, w)| w.imports.values())
313        .flat_map(as_interface)
314        .collect()
315}
316
317fn as_interface(wi: &wit_parser::WorldItem) -> Option<wit_parser::InterfaceId> {
318    match wi {
319        wit_parser::WorldItem::Interface { id, .. } => Some(*id),
320        _ => None,
321    }
322}
323
324fn one_import(wasm: &DecodedWasm, name: &str) -> anyhow::Result<Vec<wit_parser::InterfaceId>> {
325    let id = wasm
326        .resolve()
327        .interfaces
328        .iter()
329        .find(|i| i.1.name == Some(name.to_string()))
330        .map(|t| t.0)
331        .with_context(|| format!("interface {name} not found in component binary"))?;
332    Ok(vec![id])
333}
334
335fn read_wasm(wasm_bytes: &[u8]) -> anyhow::Result<DecodedWasm> {
336    if wasmparser::Parser::is_component(wasm_bytes) {
337        wit_component::decode(wasm_bytes)
338    } else {
339        let (wasm, bindgen) = wit_component::metadata::decode(wasm_bytes)?;
340        if wasm.is_none() {
341            anyhow::bail!(
342                "input is a core wasm module with no `component-type*` \
343                    custom sections meaning that there is no WIT information; \
344                    is the information not embedded or is this supposed \
345                    to be a component?"
346            )
347        }
348        Ok(DecodedWasm::Component(bindgen.resolve, bindgen.world))
349    }
350}
351
352fn importize(decoded: DecodedWasm, out_world_name: Option<&String>) -> anyhow::Result<DecodedWasm> {
353    let (mut resolve, world_id) = match decoded {
354        DecodedWasm::Component(resolve, world) => (resolve, world),
355        DecodedWasm::WitPackage(resolve, id) => {
356            let world = resolve.select_world(&[id], None)?;
357            (resolve, world)
358        }
359    };
360
361    resolve
362        .importize(world_id, out_world_name.cloned())
363        .context("failed to move world exports to imports")?;
364
365    Ok(DecodedWasm::Component(resolve, world_id))
366}
