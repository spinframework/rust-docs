107    /// Sets a [`OutboundHttpInterceptor`] for this instance.

111        &mut self,
112        interceptor: impl OutboundHttpInterceptor + 'static,
113    ) -> anyhow::Result<()> {
114        if self.request_interceptor.is_some() {
115            anyhow::bail!("set_request_interceptor can only be called once");
116        }
117        self.request_interceptor = Some(Arc::new(interceptor));
118        Ok(())
119    }

127/// SelfRequestOrigin indicates the base URI to use for "self" requests.
128#[derive(Clone, Debug)]
129pub struct SelfRequestOrigin {
130    pub scheme: Scheme,
131    pub authority: Authority,
132}
133
134impl SelfRequestOrigin {
135    pub fn create(scheme: Scheme, auth: &str) -> anyhow::Result<Self> {
136        Ok(SelfRequestOrigin {
137            scheme,
138            authority: auth
139                .parse()
140                .with_context(|| format!("address '{auth}' is not a valid authority"))?,
141        })
142    }
143
144    pub fn from_uri(uri: &Uri) -> anyhow::Result<Self> {
145        Ok(Self {
146            scheme: uri.scheme().context("URI missing scheme")?.clone(),
147            authority: uri.authority().context("URI missing authority")?.clone(),
148        })
149    }
150
151    fn into_uri(self, path_and_query: Option<PathAndQuery>) -> Uri {
152        let mut parts = Parts::default();
153        parts.scheme = Some(self.scheme);
154        parts.authority = Some(self.authority);
155        parts.path_and_query = path_and_query;
156        Uri::from_parts(parts).unwrap()
157    }
158
159    fn use_tls(&self) -> bool {
160        self.scheme == Scheme::HTTPS
161    }
162
163    fn host_header(&self) -> HeaderValue {
164        HeaderValue::from_str(self.authority.as_str()).unwrap()
165    }
166}
167
168impl std::fmt::Display for SelfRequestOrigin {
169    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
170        write!(f, "{}://{}", self.scheme, self.authority)
171    }
