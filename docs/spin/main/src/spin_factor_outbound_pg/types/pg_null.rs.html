1use anyhow::Result;
2use tokio_postgres::types::{ToSql, Type};
3
4/// Although the Postgres crate converts Rust Option::None to Postgres NULL,
5/// it enforces the type of the Option as it does so. (For example, trying to
6/// pass an Option::<i32>::None to a VARCHAR column fails conversion.) As we
7/// do not know expected column types, we instead use a "neutral" custom type
8/// which allows conversion to any type but always tells the Postgres crate to
9/// treat it as a SQL NULL.
10pub struct PgNull;
11
12impl ToSql for PgNull {
13    fn to_sql(
14        &self,
15        _ty: &Type,
16        _out: &mut tokio_postgres::types::private::BytesMut,
17    ) -> Result<tokio_postgres::types::IsNull, Box<dyn std::error::Error + Sync + Send>>
18    where
19        Self: Sized,
20    {
21        Ok(tokio_postgres::types::IsNull::Yes)
22    }
23
24    fn accepts(_ty: &Type) -> bool
25    where
26        Self: Sized,
27    {
28        true
29    }
30
31    fn to_sql_checked(
32        &self,
33        _ty: &Type,
34        _out: &mut tokio_postgres::types::private::BytesMut,
35    ) -> Result<tokio_postgres::types::IsNull, Box<dyn std::error::Error + Sync + Send>> {
36        Ok(tokio_postgres::types::IsNull::Yes)
37    }
38}
39
40impl std::fmt::Debug for PgNull {
41    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
42        f.debug_struct("NULL").finish()
43    }
44}
