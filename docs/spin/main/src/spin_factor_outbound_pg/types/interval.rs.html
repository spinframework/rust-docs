36impl FromSql<'_> for Interval {
37    fn from_sql(
38        _ty: &Type,
39        raw: &'_ [u8],
40    ) -> std::result::Result<Self, Box<dyn std::error::Error + Sync + Send>> {
41        const EXPECTED_LEN: usize = size_of::<i64>() + size_of::<i32>() + size_of::<i32>();
42
43        if raw.len() != EXPECTED_LEN {
44            return Err(Box::new(IntervalLengthError));
45        }
46
47        let (micro_bytes, rest) = raw.split_at(size_of::<i64>());
48        let (day_bytes, rest) = rest.split_at(size_of::<i32>());
49        let month_bytes = rest;
50        let months = i32::from_be_bytes(month_bytes.try_into().unwrap());
51        let days = i32::from_be_bytes(day_bytes.try_into().unwrap());
52        let micros = i64::from_be_bytes(micro_bytes.try_into().unwrap());
53
54        Ok(Self(v4::Interval {
55            micros,
56            days,
57            months,
58        }))
59    }
60
61    fn accepts(ty: &Type) -> bool {
62        matches!(ty, &Type::INTERVAL)
63    }
64}
