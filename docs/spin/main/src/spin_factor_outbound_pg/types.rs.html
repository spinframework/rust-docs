22pub fn convert_data_type(pg_type: &Type) -> DbDataType {
23    match *pg_type {
24        Type::BOOL => DbDataType::Boolean,
25        Type::BYTEA => DbDataType::Binary,
26        Type::FLOAT4 => DbDataType::Floating32,
27        Type::FLOAT8 => DbDataType::Floating64,
28        Type::INT2 => DbDataType::Int16,
29        Type::INT4 => DbDataType::Int32,
30        Type::INT8 => DbDataType::Int64,
31        Type::TEXT | Type::VARCHAR | Type::BPCHAR => DbDataType::Str,
32        Type::TIMESTAMP | Type::TIMESTAMPTZ => DbDataType::Datetime,
33        Type::DATE => DbDataType::Date,
34        Type::TIME => DbDataType::Time,
35        Type::UUID => DbDataType::Uuid,
36        Type::JSONB => DbDataType::Jsonb,
37        Type::NUMERIC => DbDataType::Decimal,
38        Type::INT4_RANGE => DbDataType::RangeInt32,
39        Type::INT8_RANGE => DbDataType::RangeInt64,
40        Type::NUM_RANGE => DbDataType::RangeDecimal,
41        Type::INT4_ARRAY => DbDataType::ArrayInt32,
42        Type::INT8_ARRAY => DbDataType::ArrayInt64,
43        Type::NUMERIC_ARRAY => DbDataType::ArrayDecimal,
44        Type::TEXT_ARRAY | Type::VARCHAR_ARRAY | Type::BPCHAR_ARRAY => DbDataType::ArrayStr,
45        Type::INTERVAL => DbDataType::Interval,
46        _ => {
47            tracing::debug!("Couldn't convert Postgres type {} to WIT", pg_type.name(),);
48            DbDataType::Other(pg_type.name().to_owned())
49        }
50    }
51}

53fn db_value<'a, T: FromSql<'a>>(
54    row: &'a Row,
55    index: usize,
56    convert_fn: impl Fn(T) -> DbValue,
57) -> anyhow::Result<DbValue> {
58    let value: Option<T> = row.try_get(index)?;
59    Ok(match value {
60        Some(v) => convert_fn(v),
61        None => DbValue::DbNull,
62    })
63}

65fn map_db_value<'a, T: FromSql<'a>, W>(
66    row: &'a Row,
67    index: usize,
68    ctor: impl Fn(W) -> DbValue,
69    convert_fn: impl Fn(T) -> W,
70) -> anyhow::Result<DbValue> {
71    let value: Option<T> = row.try_get(index)?;
72    Ok(match value {
73        Some(v) => ctor(convert_fn(v)),
74        None => DbValue::DbNull,
75    })
76}

91pub fn convert_entry(row: &Row, index: usize) -> anyhow::Result<DbValue> {
92    let column = &row.columns()[index];
93    match column.type_() {
94        &Type::BOOL => db_value(row, index, DbValue::Boolean),
95        &Type::BYTEA => db_value(row, index, DbValue::Binary),
96        &Type::FLOAT4 => db_value(row, index, DbValue::Floating32),
97        &Type::FLOAT8 => db_value(row, index, DbValue::Floating64),
98        &Type::INT2 => db_value(row, index, DbValue::Int16),
99        &Type::INT4 => db_value(row, index, DbValue::Int32),
100        &Type::INT8 => db_value(row, index, DbValue::Int64),
101        &Type::TEXT | &Type::VARCHAR | &Type::BPCHAR => db_value(row, index, DbValue::Str),
102        &Type::TIMESTAMP | &Type::TIMESTAMPTZ => {
103            try_map_db_value(row, index, DbValue::Datetime, datetime_pg_to_wit)
104        }
105        &Type::DATE => try_map_db_value(row, index, DbValue::Date, date_pg_to_wit),
106        &Type::TIME => try_map_db_value(row, index, DbValue::Time, time_pg_to_wit),
107        &Type::UUID => map_db_value(row, index, DbValue::Uuid, |v: uuid::Uuid| v.to_string()),
108        &Type::JSONB => try_map_db_value(row, index, DbValue::Jsonb, jsonb_pg_to_wit),
109        &Type::NUMERIC => map_db_value(row, index, DbValue::Decimal, |v: rust_decimal::Decimal| {
110            v.to_string()
111        }),
112        &Type::INT4_RANGE => map_db_value(row, index, DbValue::RangeInt32, range_pg_to_wit),
113        &Type::INT8_RANGE => map_db_value(row, index, DbValue::RangeInt64, range_pg_to_wit),
114        &Type::NUM_RANGE => {
115            map_db_value(row, index, DbValue::RangeDecimal, decimal_range_pg_to_wit)
116        }
117        &Type::INT4_ARRAY => db_value(row, index, DbValue::ArrayInt32),
118        &Type::INT8_ARRAY => db_value(row, index, DbValue::ArrayInt64),
119        &Type::NUMERIC_ARRAY => {
120            map_db_value(row, index, DbValue::ArrayDecimal, decimal_array_pg_to_wit)
121        }
122        &Type::TEXT_ARRAY | &Type::VARCHAR_ARRAY | &Type::BPCHAR_ARRAY => {
123            db_value(row, index, DbValue::ArrayStr)
124        }
125        &Type::INTERVAL => map_db_value(row, index, DbValue::Interval, |v: Interval| v.into()),
126        t => {
127            tracing::debug!(
128                "Couldn't convert Postgres type {} in column {}",
129                t.name(),
130                column.name()
131            );
132            map_db_value(row, index, DbValue::Unsupported, |v: Other| v.into())
133        }
134    }
135}

137pub fn to_sql_parameter(value: &ParameterValue) -> anyhow::Result<Box<dyn ToSql + Send + Sync>> {
138    match value {
139        ParameterValue::Boolean(v) => Ok(Box::new(*v)),
140        ParameterValue::Int32(v) => Ok(Box::new(*v)),
141        ParameterValue::Int64(v) => Ok(Box::new(*v)),
142        ParameterValue::Int8(v) => Ok(Box::new(*v)),
143        ParameterValue::Int16(v) => Ok(Box::new(*v)),
144        ParameterValue::Floating32(v) => Ok(Box::new(*v)),
145        ParameterValue::Floating64(v) => Ok(Box::new(*v)),
146        ParameterValue::Str(v) => Ok(Box::new(v.clone())),
147        ParameterValue::Binary(v) => Ok(Box::new(v.clone())),
148        ParameterValue::Date(v) => Ok(Box::new(date_wit_to_pg(v)?)),
149        ParameterValue::Time(v) => Ok(Box::new(time_wit_to_pg(v)?)),
150        ParameterValue::Datetime(v) => Ok(Box::new(datetime_wit_to_pg(v)?)),
151        ParameterValue::Timestamp(v) => Ok(Box::new(timestamp_wit_to_pg(*v)?)),
152        ParameterValue::Uuid(v) => Ok(Box::new(uuid_wit_to_pg(v)?)),
153        ParameterValue::Jsonb(v) => Ok(Box::new(jsonb_wit_to_pg(v)?)),
154        ParameterValue::Decimal(v) => Ok(Box::new(decimal_wit_to_pg(v)?)),
155        ParameterValue::RangeInt32(v) => Ok(Box::new(range_wit_to_pg(*v))),
156        ParameterValue::RangeInt64(v) => Ok(Box::new(range_wit_to_pg(*v))),
157        ParameterValue::RangeDecimal(v) => Ok(Box::new(decimal_range_wit_to_pg(v)?)),
158        ParameterValue::ArrayInt32(vs) => Ok(Box::new(vs.to_owned())),
159        ParameterValue::ArrayInt64(vs) => Ok(Box::new(vs.to_owned())),
160        ParameterValue::ArrayDecimal(vs) => Ok(Box::new(decimal_array_wit_to_pg(vs)?)),
161        ParameterValue::ArrayStr(vs) => Ok(Box::new(vs.to_owned())),
162        ParameterValue::Interval(v) => Ok(Box::new(Interval(*v))),
163        ParameterValue::DbNull => Ok(Box::new(PgNull)),
164    }
165}

167// The logic for "vector of ParameterValue to vector of &dyn ToSql" is
168// used in multiple places, but needs to be broken into two functions
169// because the return value of the first (the Vec<Box>) needs to be kept
170// around to provide an owner for the refs.
171#[allow(clippy::result_large_err)]
172pub fn to_sql_parameters(
173    params: Vec<ParameterValue>,
174) -> Result<Vec<Box<dyn ToSql + Send + Sync>>, v4::Error> {
175    params
176        .iter()
177        .map(to_sql_parameter)
178        .collect::<Result<Vec<_>>>()
179        .map_err(|e| v4::Error::BadParameter(format!("{e:?}")))
180}
