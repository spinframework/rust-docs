22/// If present, run the build command of each component.
23pub async fn build(
24    manifest_file: &Path,
25    profile: Option<&str>,
26    component_ids: &[String],
27    target_checks: TargetChecking,
28    wit_generation: GenerateDependencyWits,
29    cache_root: Option<PathBuf>,
30) -> Result<()> {
31    let build_info = component_build_configs(manifest_file, profile)
32        .await
33        .with_context(|| {
34            format!(
35                "Cannot read manifest file from {}",
36                quoted_path(manifest_file)
37            )
38        })?;
39    let app_dir = parent_dir(manifest_file)?;
40
41    let components_to_build = components_to_build(component_ids, build_info.components())?;
42
43    if wit_generation.generate() {
44        let wit_gen_errs = regenerate_wits(&components_to_build, &app_dir).await;
45        if !wit_gen_errs.is_empty() {
46            terminal::warn!("One or more components specified dependencies for which Spin couldn't generate import interfaces.");
47            eprintln!(
48                "If these components rely on Spin-generated interfaces they may fail to build."
49            );
50            eprintln!(
51                "Otherwise, to skip interface generation, use the --skip-generate-wits flag."
52            );
53            eprintln!("Error details:");
54            for (component, err) in wit_gen_errs {
55                terminal::einfo!("{component}:", "{err:#}");
56            }
57        }
58    }
59
60    let build_result = build_components(components_to_build, &app_dir);
61
62    // Emit any required warnings now, so that they don't bury any errors.
63    if let Some(e) = build_info.load_error() {
64        // The manifest had errors. We managed to attempt a build anyway, but we want to
65        // let the user know about them.
66        terminal::warn!("The manifest has errors not related to the Wasm component build. Error details:\n{e:#}");
67        // Checking deployment targets requires a healthy manifest (because trigger types etc.),
68        // if any of these were specified, warn they are being skipped.
69        let should_have_checked_targets =
70            target_checks.check() && build_info.has_deployment_targets();
71        if should_have_checked_targets {
72            terminal::warn!(
73                "The manifest error(s) prevented Spin from checking the deployment targets."
74            );
75        }
76    }
77
78    // If the build failed, exit with an error at this point.
79    build_result?;
80
81    if let Err(e) = save_last_build_profile(&app_dir, profile) {
82        tracing::warn!("Failed to save build profile: {e:?}");
83    }
84
85    let Some(manifest) = build_info.manifest() else {
86        // We can't proceed to checking (because that needs a full healthy manifest), and we've
87        // already emitted any necessary warning, so quit.
88        return Ok(());
89    };
90
91    if target_checks.check() {
92        let application = spin_environments::ApplicationToValidate::new(
93            manifest.clone(),
94            profile,
95            manifest_file.parent().unwrap(),
96        )
97        .await
98        .context("unable to load application for checking against deployment targets")?;
99        let target_validation = spin_environments::validate_application_against_environment_ids(
100            &application,
101            build_info.deployment_targets(),
102            cache_root.clone(),
103            &app_dir,
104        )
105        .await
106        .context("unable to check if the application is compatible with deployment targets")?;
107
108        if !target_validation.is_ok() {
109            for error in target_validation.errors() {
110                terminal::error!("{error}");
111            }
112            anyhow::bail!("All components built successfully, but one or more was incompatible with one or more of the deployment targets.");
113        }
114    }
115
116    Ok(())
117}

119/// Run all component build commands, using the default options (build all
120/// components, perform target checking). We run a "default build" in several
121/// places and this centralises the logic of what such a "default build" means.
122pub async fn build_default(
123    manifest_file: &Path,
124    profile: Option<&str>,
125    cache_root: Option<PathBuf>,
126) -> Result<()> {

128        manifest_file,
129        profile,

134    )
135    .await

138fn components_to_build(
139    component_ids: &[String],
140    components: Vec<ComponentBuildInfo>,
141) -> anyhow::Result<Vec<ComponentBuildInfo>> {
142    let components_to_build = if component_ids.is_empty() {
143        components
144    } else {
145        let all_ids: HashSet<_> = components.iter().map(|c| &c.id).collect();
146        let unknown_component_ids: Vec<_> = component_ids
147            .iter()
148            .filter(|id| !all_ids.contains(id))
149            .map(|s| s.as_str())
150            .collect();
151
152        if !unknown_component_ids.is_empty() {
153            bail!("Unknown component(s) {}", unknown_component_ids.join(", "));
154        }
155
156        components
157            .into_iter()
158            .filter(|c| component_ids.contains(&c.id))
159            .collect()
160    };
161
162    Ok(components_to_build)
163}

165#[must_use]
166async fn regenerate_wits(
167    components_to_build: &[ComponentBuildInfo],
168    app_root: &Path,
169) -> Vec<(String, anyhow::Error)> {
170    let mut errors = vec![];
171
172    for component in components_to_build {
173        let component_dir = match component.build.as_ref().and_then(|b| b.workdir.as_ref()) {
174            None => app_root.to_owned(),
175            Some(d) => app_root.join(d),
176        };
177        let dest_file = component_dir.join("spin-dependencies.wit");
178        let extract_result = spin_dependency_wit::extract_wits_into(
179            component.dependencies.inner.iter(),
180            app_root,
181            dest_file,
182        )
183        .await;
184        if let Err(e) = extract_result {
185            errors.push((component.id.clone(), e));
186        }
187    }
188
189    errors
190}

192fn build_components(
193    components_to_build: Vec<ComponentBuildInfo>,
194    app_dir: &Path,
195) -> anyhow::Result<()> {
196    if components_to_build.iter().all(|c| c.build.is_none()) {
197        println!("None of the components have a build command.");
198        println!("For information on specifying a build command, see https://spinframework.dev/build#setting-up-for-spin-build.");
199        return Ok(());
200    }
201
202    // If dependencies are being built as part of `spin build`, we would like
203    // them to be rebuilt earlier (e.g. so that consumers using the binary as a source
204    // of type information see the latest interface).
205    let (components_to_build, has_cycle) = sort(components_to_build);
206
207    if has_cycle {
208        tracing::debug!("There is a dependency cycle among components. Spin cannot guarantee to build dependencies before consumers.");
209    }
210
211    components_to_build
212        .into_iter()
213        .map(|c| build_component(c, app_dir))
214        .collect::<Result<Vec<_>, _>>()?;
215
216    terminal::step!("Finished", "building all Spin components");
217    Ok(())
218}

220/// Run the build command of the component.
221fn build_component(build_info: ComponentBuildInfo, app_dir: &Path) -> Result<()> {
222    match build_info.build {
223        Some(b) => {
224            let command_count = b.commands().len();
225
226            if command_count > 1 {
227                terminal::step!(
228                    "Building",
229                    "component {} ({} commands)",
230                    build_info.id,
231                    command_count
232                );
233            }
234
235            for (index, command) in b.commands().enumerate() {
236                if command_count > 1 {
237                    terminal::step!(
238                        "Running build step",
239                        "{}/{} for component {} with '{}'",
240                        index + 1,
241                        command_count,
242                        build_info.id,
243                        command
244                    );
245                } else {
246                    terminal::step!("Building", "component {} with `{}`", build_info.id, command);
247                }
248
249                let workdir = construct_workdir(app_dir, b.workdir.as_ref())?;
250                if b.workdir.is_some() {
251                    println!("Working directory: {}", quoted_path(&workdir));
252                }
253
254                let exit_status = Exec::shell(command)
255                    .cwd(workdir)
256                    .stdout(Redirection::None)
257                    .stderr(Redirection::None)
258                    .stdin(Redirection::None)
259                    .popen()
260                    .map_err(|err| {
261                        anyhow!(
262                            "Cannot spawn build process '{:?}' for component {}: {}",
263                            &b.command,
264                            build_info.id,
265                            err
266                        )
267                    })?
268                    .wait()?;
269
270                if !exit_status.success() {
271                    bail!(
272                        "Build command for component {} failed with status {:?}",
273                        build_info.id,
274                        exit_status,
275                    );
276                }
277            }
278
279            Ok(())
280        }
281        _ => Ok(()),
282    }
283}

285/// Constructs the absolute working directory in which to run the build command.
286fn construct_workdir(app_dir: &Path, workdir: Option<impl AsRef<Path>>) -> Result<PathBuf> {
287    let mut cwd = app_dir.to_owned();
288
289    if let Some(workdir) = workdir {
290        // Using `Path::has_root` as `is_relative` and `is_absolute` have
291        // surprising behavior on Windows, see:
292        // https://doc.rust-lang.org/std/path/struct.Path.html#method.is_absolute
293        if workdir.as_ref().has_root() {
294            bail!("The workdir specified in the application file must be relative.");
295        }
296        cwd.push(workdir);
297    }
298
299    Ok(cwd)
300}

302#[derive(Clone)]
303struct SortableBuildInfo {
304    source: Option<String>,
305    local_dependency_paths: Vec<String>,
306    build_info: ComponentBuildInfo,
307}
308
309impl From<&ComponentBuildInfo> for SortableBuildInfo {
310    fn from(value: &ComponentBuildInfo) -> Self {
311        fn local_dep_path(dep: &v2::ComponentDependency) -> Option<String> {
312            match dep {
313                v2::ComponentDependency::Local { path, .. } => Some(path.display().to_string()),
314                _ => None,
315            }
316        }
317
318        let source = match value.source.as_ref() {
319            Some(spin_manifest::schema::v2::ComponentSource::Local(path)) => Some(path.clone()),
320            _ => None,
321        };
322        let local_dependency_paths = value
323            .dependencies
324            .inner
325            .values()
326            .filter_map(local_dep_path)
327            .collect();
328
329        Self {
330            source,
331            local_dependency_paths,
332            build_info: value.clone(),
333        }
334    }
335}
336
337impl std::hash::Hash for SortableBuildInfo {
338    fn hash<H: std::hash::Hasher>(&self, state: &mut H) {
339        self.build_info.id.hash(state);
340        self.source.hash(state);
341        self.local_dependency_paths.hash(state);
342    }
343}
344
345impl PartialEq for SortableBuildInfo {
346    fn eq(&self, other: &Self) -> bool {
347        self.build_info.id == other.build_info.id
348            && self.source == other.source
349            && self.local_dependency_paths == other.local_dependency_paths
350    }
351}

355/// Topo sort by local path dependency. Second result is if there was a cycle.
356fn sort(components: Vec<ComponentBuildInfo>) -> (Vec<ComponentBuildInfo>, bool) {
357    let sortables = components
358        .iter()
359        .map(SortableBuildInfo::from)
360        .collect::<Vec<_>>();
361    let mut sorter = topological_sort::TopologicalSort::<SortableBuildInfo>::new();
362
363    for s in &sortables {
364        sorter.insert(s.clone());
365    }
366
367    for s1 in &sortables {
368        for dep in &s1.local_dependency_paths {
369            for s2 in &sortables {
370                if s2.source.as_ref().is_some_and(|src| src == dep) {
371                    // s1 depends on s2
372                    sorter.add_link(topological_sort::DependencyLink {
373                        prec: s2.clone(),
374                        succ: s1.clone(),
375                    });
376                }
377            }
378        }
379    }
380
381    let result = sorter.map(|s| s.build_info).collect::<Vec<_>>();
382
383    // We shouldn't refuse to build if a cycle occurs, so return the original order to allow
384    // stuff to proceed.  (We could be smarter about this, but really it's a pathological situation
385    // and we don't need to bust a gut over it.)
386    if result.len() == components.len() {
387        (result, false)
388    } else {
389        (components, true)
390    }
391}

393/// Saves the build profile to the "last build profile" file.
394pub fn save_last_build_profile(app_dir: &Path, profile: Option<&str>) -> anyhow::Result<()> {
395    let app_stash_dir = app_dir.join(".spin");
396    let last_build_profile_file = app_stash_dir.join(LAST_BUILD_PROFILE_FILE);
397
398    // This way, if the user never uses build profiles, they won't see a
399    // weird savefile that they have no idea what it is.
400    if profile.is_none() && !last_build_profile_file.exists() {
401        return Ok(());
402    }
403
404    std::fs::create_dir_all(&app_stash_dir)?;
405    std::fs::write(
406        &last_build_profile_file,
407        profile.unwrap_or(LAST_BUILD_ANON_VALUE),
408    )?;
409
410    Ok(())
411}
412
413/// Reads the last build profile from the "last build profile" file.
414pub fn read_last_build_profile(app_dir: &Path) -> anyhow::Result<Option<String>> {
415    let app_stash_dir = app_dir.join(".spin");
416    let last_build_profile_file = app_stash_dir.join(LAST_BUILD_PROFILE_FILE);
417    if !last_build_profile_file.exists() {
418        return Ok(None);
419    }
420
421    let last_build_str = std::fs::read_to_string(&last_build_profile_file)?;
422
423    if last_build_str == LAST_BUILD_ANON_VALUE {
424        Ok(None)
425    } else {
426        Ok(Some(last_build_str))
427    }
428}

430/// Prints a warning to stderr if the given profile is not the same
431/// as the most recent build in the given application directory.
432pub fn warn_if_not_latest_build(manifest_path: &Path, profile: Option<&str>) {
433    let Some(app_dir) = manifest_path.parent() else {
434        return;
435    };
436
437    let latest_build = match read_last_build_profile(app_dir) {
438        Ok(profile) => profile,
439        Err(e) => {
440            tracing::warn!(
441                "Failed to read last build profile: using anonymous profile. Error was {e:?}"
442            );
443            None
444        }
445    };
446
447    if profile != latest_build.as_deref() {
448        let profile_opt = match profile {
449            Some(p) => format!(" --profile {p}"),
450            None => "".to_string(),
451        };
452        terminal::warn!("You built a different profile more recently than the one you are running. If the app appears to be behaving like an older version then run `spin up --build{profile_opt}`.");
453    }
454}

456/// Specifies target environment checking behaviour
457pub enum TargetChecking {
458    /// The build should check that all components are compatible with all target environments.
459    Check,
460    /// The build should not check target environments.
461    Skip,
462}
463
464impl TargetChecking {
465    /// Should the build check target environments?
466    fn check(&self) -> bool {
467        matches!(self, Self::Check)
468    }
469}

471/// Specifies dependency WIT generation behaviour
472pub enum GenerateDependencyWits {
473    /// The build should generate WITs for component dependencies.
474    Generate,
475    /// The build should not generate WITs.
476    Skip,
477}
478
479impl GenerateDependencyWits {
480    /// Should the build generate dependency WITs?
481    fn generate(&self) -> bool {
482        matches!(self, Self::Generate)
483    }
