47    // Load the manifest file (spin.toml) at the given path into a LockedApp,
48    // preparing all its content for execution.
49    pub async fn load_file(&self, path: impl AsRef<Path>) -> Result<LockedApp> {
50        // Parse manifest
51        let path = path.as_ref();
52        let manifest = spin_manifest::manifest_from_file(path).with_context(|| {
53            format!(
54                "Failed to read Spin app manifest from {}",
55                quoted_path(path)
56            )
57        })?;
58        let mut locked = self
59            .load_manifest(manifest)
60            .await
61            .with_context(|| format!("Failed to load Spin app from {}", quoted_path(path)))?;
62
63        // Set origin metadata
64        locked
65            .metadata
66            .insert("origin".into(), file_url(path)?.into());
67
68        Ok(locked)
69    }

71    // Load the given manifest into a LockedApp, ready for execution.
72    pub(crate) async fn load_manifest(&self, mut manifest: AppManifest) -> Result<LockedApp> {
73        spin_manifest::normalize::normalize_manifest(&mut manifest);
74
75        manifest.validate_dependencies()?;
76
77        let AppManifest {
78            spin_manifest_version: _,
79            application,
80            variables,
81            triggers,
82            components,
83        } = manifest;
84
85        let metadata = locked_metadata(application, triggers.keys().cloned())?;
86
87        let app_requires_service_chaining = components.values().any(requires_service_chaining);
88
89        let variables = variables
90            .into_iter()
91            .map(|(name, v)| Ok((name.to_string(), locked_variable(v)?)))
92            .collect::<Result<_>>()?;
93
94        let triggers = triggers
95            .into_iter()
96            .flat_map(|(trigger_type, configs)| {
97                configs
98                    .into_iter()
99                    .map(|trigger| locked_trigger(trigger_type.clone(), trigger))
100                    .collect::<Vec<_>>()
101            })
102            .collect::<Result<Vec<_>>>()?;
103
104        let sloth_guard = warn_if_component_load_slothful();
105
106        // Load all components concurrently
107        let components = try_join_all(components.into_iter().map(|(id, c)| async move {
108            self.load_component(&id, c)
109                .await
110                .with_context(|| format!("Failed to load component `{id}`"))
111        }))
112        .await?;
113
114        let mut host_requirements = ValuesMapBuilder::new();
115        if app_requires_service_chaining {
116            host_requirements.string(
117                spin_locked_app::locked::SERVICE_CHAINING_KEY,
118                spin_locked_app::locked::HOST_REQ_REQUIRED,
119            );
120        }
121        let host_requirements = host_requirements.build();
122
123        let mut must_understand = vec![];
124        if !host_requirements.is_empty() {
125            must_understand.push(spin_locked_app::locked::MustUnderstand::HostRequirements);
126        }
127
128        drop(sloth_guard);
129
130        Ok(LockedApp {
131            spin_lock_version: Default::default(),
132            metadata,
133            must_understand,
134            host_requirements,
135            variables,
136            triggers,
137            components,
138        })
139    }

141    // Load the given component into a LockedComponent, ready for execution.
142    async fn load_component(
143        &self,
144        id: &KebabId,
145        component: v2::Component,
146    ) -> Result<LockedComponent> {
147        let allowed_outbound_hosts = component
148            .normalized_allowed_outbound_hosts()
149            .context("`allowed_http_hosts` is malformed")?;
150        spin_factor_outbound_networking::AllowedHostsConfig::validate(&allowed_outbound_hosts)
151            .context("`allowed_outbound_hosts` is malformed")?;
152
153        let metadata = ValuesMapBuilder::new()
154            .string("description", component.description)
155            .string_array("allowed_outbound_hosts", allowed_outbound_hosts)
156            .string_array("key_value_stores", component.key_value_stores)
157            .string_array("databases", component.sqlite_databases)
158            .string_array("ai_models", component.ai_models)
159            .serializable("build", component.build)?
160            .take();
161
162        let source = self
163            .load_component_source(id, component.source.clone())
164            .await
165            .with_context(|| format!("Failed to load Wasm source {}", component.source))?;
166
167        let dependencies = self
168            .load_component_dependencies(
169                id,
170                component.dependencies_inherit_configuration,
171                &component.dependencies,
172            )
173            .await?;
174
175        let env = component.environment.into_iter().collect();
176
177        let files = if component.files.is_empty() {
178            vec![]
179        } else {
180            match &self.files_mount_strategy {
181                FilesMountStrategy::Copy(files_mount_root) => {
182                    let component_mount_root = files_mount_root.join(id.as_ref());
183                    // Copy mounted files into component mount root, concurrently
184                    try_join_all(component.files.iter().map(|f| {
185                        self.copy_file_mounts(f, &component_mount_root, &component.exclude_files)
186                    }))
187                    .await?;
188
189                    // All component files (copies) are in `component_mount_root` now
190                    vec![ContentPath {
191                        content: file_content_ref(component_mount_root)?,
192                        path: "/".into(),
193                    }]
194                }
195                FilesMountStrategy::Direct => {
196                    ensure!(
197                        component.exclude_files.is_empty(),
198                        "Cannot load a component with `exclude_files` using --direct-mounts"
199                    );
200                    let mut files = vec![];
201                    for mount in &component.files {
202                        // Validate (and canonicalize) direct mount directory
203                        files.push(self.resolve_direct_mount(mount).await?);
204                    }
205                    files
206                }
207            }
208        };
209
210        let config = component
211            .variables
212            .into_iter()
213            .map(|(k, v)| (k.into(), v))
214            .collect();
215
216        Ok(LockedComponent {
217            id: id.as_ref().into(),
218            metadata,
219            source,
220            env,
221            files,
222            config,
223            dependencies,
224        })
225    }

256    async fn load_component_dependency(
257        &self,
258        inherit_configuration: bool,
259        dependency_name: DependencyName,
260        dependency: v2::ComponentDependency,
261    ) -> Result<LockedComponentDependency> {
262        let (content, export) = match dependency {
263            v2::ComponentDependency::Version(version) => {
264                let version = semver::VersionReq::parse(&version).with_context(|| format!("Component dependency {dependency_name:?} specifies an invalid semantic version requirement ({version:?}) for its package version"))?;
265
266                // This `unwrap()` should be OK because we've already validated
267                // this form of dependency requires a package name, i.e. the
268                // dependency name is not a kebab id.
269                let package = dependency_name.package().unwrap();
270
271                let content = self.load_registry_source(None, package, &version).await?;
272                (content, None)
273            }
274            v2::ComponentDependency::Package {
275                version,
276                registry,
277                package,
278                export,
279            } => {
280                let version = semver::VersionReq::parse(&version).with_context(|| format!("Component dependency {dependency_name:?} specifies an invalid semantic version requirement ({version:?}) for its package version"))?;
281
282                let package = match package {
283                    Some(package) => {
284                        package.parse().with_context(|| format!("Component dependency {dependency_name:?} specifies an invalid package name ({package:?})"))?
285                    }
286                    None => {
287                        // This `unwrap()` should be OK because we've already validated
288                        // this form of dependency requires a package name, i.e. the
289                        // dependency name is not a kebab id.
290                        dependency_name
291                            .package()
292                            .cloned()
293                            .unwrap()
294                    }
295                };
296
297                let registry = match registry {
298                    Some(registry) => {
299                        registry
300                            .parse()
301                            .map(Some)
302                            .with_context(|| format!("Component dependency {dependency_name:?} specifies an invalid registry name ({registry:?})"))?
303                    }
304                    None => None,
305                };
306
307                let content = self
308                    .load_registry_source(registry.as_ref(), &package, &version)
309                    .await?;
310                (content, export)
311            }
312            v2::ComponentDependency::Local { path, export } => {
313                let content = file_content_ref(self.app_root.join(path))?;
314                (content, export)
315            }
316            v2::ComponentDependency::HTTP {
317                url,
318                digest,
319                export,
320            } => {
321                let content = self.load_http_source(&url, &digest).await?;
322                (content, export)
323            }
324        };
325
326        Ok(LockedComponentDependency {
327            source: LockedComponentSource {
328                content_type: "application/wasm".into(),
329                content,
330            },
331            export,
332            inherit: if inherit_configuration {
333                locked::InheritConfiguration::All
334            } else {
335                locked::InheritConfiguration::Some(vec![])
336            },
337        })
338    }

340    // Load a Wasm source from the given ContentRef and update the source
341    // URL with an absolute path to the content.
342    async fn load_component_source(
343        &self,
344        component_id: &KebabId,
345        source: v2::ComponentSource,
346    ) -> Result<LockedComponentSource> {
347        let content = match source {
348            v2::ComponentSource::Local(path) => file_content_ref(self.app_root.join(path))?,
349            v2::ComponentSource::Remote { url, digest } => {
350                self.load_http_source(&url, &digest).await?
351            }
352            v2::ComponentSource::Registry {
353                registry,
354                package,
355                version,
356            } => {
357                let version = semver::Version::parse(&version).with_context(|| format!("Component {component_id} specifies an invalid semantic version ({version:?}) for its package version"))?;
358                let version_req = format!("={version}").parse().expect("version");
359
360                self.load_registry_source(registry.as_ref(), &package, &version_req)
361                    .await?
362            }
363        };
364        Ok(LockedComponentSource {
365            content_type: "application/wasm".into(),
366            content,
367        })
368    }

370    // Load a Wasm source from the given HTTP ContentRef source URL and
371    // return a ContentRef an absolute path to the local copy.
372    async fn load_http_source(&self, url: &str, digest: &str) -> Result<ContentRef> {
373        ensure!(
374            digest.starts_with("sha256:"),
375            "invalid `digest` {digest:?}; must start with 'sha256:'"
376        );
377        let path = if let Ok(cached_path) = self.cache.wasm_file(digest) {
378            cached_path
379        } else {
380            let _loading_permit = self.file_loading_permits.acquire().await?;
381
382            self.cache.ensure_dirs().await?;
383            let dest = self.cache.wasm_path(digest);
384            verified_download(
385                url,
386                digest,
387                &dest,
388                crate::http::DestinationConvention::ContentIndexed,
389            )
390            .await
391            .with_context(|| format!("Error fetching source URL {url:?}"))?;
392            dest
393        };
394        file_content_ref(path)
395    }

397    async fn load_registry_source(
398        &self,
399        registry: Option<&wasm_pkg_client::Registry>,
400        package: &wasm_pkg_client::PackageRef,
401        version: &semver::VersionReq,
402    ) -> Result<ContentRef> {
403        let mut client_config = wasm_pkg_client::Config::global_defaults().await?;
404
405        if let Some(registry) = registry.cloned() {
406            let mapping = wasm_pkg_client::RegistryMapping::Registry(registry);
407            client_config.set_package_registry_override(package.clone(), mapping);
408        }
409        let pkg_loader = wasm_pkg_client::Client::new(client_config);
410
411        let mut releases = pkg_loader.list_all_versions(package).await.map_err(|e| {
412            if matches!(e, wasm_pkg_client::Error::NoRegistryForNamespace(_)) && registry.is_none() {
413                anyhow!("No default registry specified for wasm-pkg-loader. Create a default config, or set `registry` for package {package:?}")
414            } else {
415                e.into()
416            }
417        })?;
418
419        releases.sort();
420
421        let release_version = releases
422            .iter()
423            .rev()
424            .find(|release| version.matches(&release.version) && !release.yanked)
425            .with_context(|| format!("No matching version found for {package} {version}",))?;
426
427        let release = pkg_loader
428            .get_release(package, &release_version.version)
429            .await?;
430
431        let digest = match &release.content_digest {
432            wasm_pkg_client::ContentDigest::Sha256 { hex } => format!("sha256:{hex}"),
433        };
434
435        let path = if let Ok(cached_path) = self.cache.wasm_file(&digest) {
436            cached_path
437        } else {
438            let mut stm = pkg_loader.stream_content(package, &release).await?;
439
440            self.cache.ensure_dirs().await?;
441            let dest = self.cache.wasm_path(&digest);
442
443            let mut file = tokio::fs::File::create(&dest).await?;
444            while let Some(block) = stm.next().await {
445                let bytes = block.context("Failed to get content from registry")?;
446                file.write_all(&bytes)
447                    .await
448                    .context("Failed to save registry content to cache")?;
449            }
450
451            dest
452        };
453
454        file_content_ref(path)
455    }

457    // Copy content(s) from the given `mount`
458    async fn copy_file_mounts(
459        &self,
460        mount: &WasiFilesMount,
461        dest_root: &Path,
462        exclude_files: &[String],
463    ) -> Result<()> {
464        match mount {
465            WasiFilesMount::Pattern(pattern) => {
466                self.copy_glob_or_path(pattern, dest_root, exclude_files)
467                    .await
468            }
469            WasiFilesMount::Placement {
470                source,
471                destination,
472            } => {
473                let src = Path::new(source);
474                let dest = dest_root.join(destination.trim_start_matches('/'));
475                self.copy_file_or_directory(src, &dest, destination, exclude_files)
476                    .await
477            }
478        }
479    }

481    // Copy files matching glob pattern or single file/directory path.
482    async fn copy_glob_or_path(
483        &self,
484        glob_or_path: &str,
485        dest_root: &Path,
486        exclude_files: &[String],
487    ) -> Result<()> {
488        let path = self.app_root.join(glob_or_path);
489        if path.exists() {
490            let dest = dest_root.join(glob_or_path);
491            if path.is_dir() {
492                // "single/dir"
493                let pattern = path.join("**/*");
494                self.copy_glob(&pattern, &self.app_root, &dest, exclude_files)
495                    .await?;
496            } else {
497                // "single/file.txt"
498                self.copy_single_file(&path, &dest, glob_or_path).await?;
499            }
500        } else if looks_like_glob_pattern(glob_or_path) {
501            // "glob/pattern/*"
502            self.copy_glob(&path, &self.app_root, dest_root, exclude_files)
503                .await?;
504        } else {
505            bail!("{glob_or_path:?} does not exist and doesn't appear to be a glob pattern");
506        }
507        Ok(())
508    }

510    // Copy a single file or entire directory from `src` to `dest`
511    async fn copy_file_or_directory(
512        &self,
513        src: &Path,
514        dest: &Path,
515        guest_dest: &str,
516        exclude_files: &[String],
517    ) -> Result<()> {
518        let src_path = self.app_root.join(src);
519        let meta = crate::fs::metadata(&src_path)
520            .await
521            .map_err(|e| explain_file_mount_source_error(e, src))?;
522        if meta.is_dir() {
523            // { source = "host/dir", destination = "guest/dir" }
524            let pattern = src_path.join("**/*");
525            self.copy_glob(&pattern, &src_path, dest, exclude_files)
526                .await?;
527        } else {
528            // { source = "host/file.txt", destination = "guest/file.txt" }
529            self.copy_single_file(&src_path, dest, guest_dest).await?;
530        }
531        Ok(())
532    }

534    // Copy files matching glob `pattern` into `dest_root`.
535    async fn copy_glob(
536        &self,
537        pattern: &Path,
538        src_prefix: &Path,
539        dest_root: &Path,
540        exclude_files: &[String],
541    ) -> Result<()> {
542        let pattern = pattern
543            .to_str()
544            .with_context(|| format!("invalid (non-utf8) file pattern {pattern:?}"))?;
545
546        let paths = glob::glob(pattern)
547            .with_context(|| format!("Failed to resolve glob pattern {pattern:?}"))?;
548
549        let exclude_patterns = exclude_files
550            .iter()
551            .map(|pattern| {
552                glob::Pattern::new(pattern)
553                    .with_context(|| format!("Invalid exclude_files glob pattern {pattern:?}"))
554            })
555            .collect::<Result<Vec<_>>>()?;
556
557        crate::fs::create_dir_all(dest_root)
558            .await
559            .with_context(|| {
560                format!(
561                    "Failed to create parent directory {}",
562                    quoted_path(&dest_root)
563                )
564            })?;
565
566        for path_res in paths {
567            let src = path_res?;
568            if !src.is_file() {
569                continue;
570            }
571
572            let Ok(app_root_path) = src.strip_prefix(&self.app_root) else {
573                bail!("{pattern} cannot be mapped because it is outside the application directory. Files must be within the application directory.");
574            };
575
576            if exclude_patterns
577                .iter()
578                .any(|pattern| pattern.matches_path(app_root_path))
579            {
580                tracing::debug!(
581                    "File {app_root_path:?} excluded by exclude_files {exclude_files:?}"
582                );
583                continue;
584            }
585
586            let relative_path = src.strip_prefix(src_prefix)?;
587            let dest = dest_root.join(relative_path);
588            self.copy_single_file(&src, &dest, &relative_path.to_string_lossy())
589                .await?;
590        }
591        Ok(())
592    }

594    // Copy a single file from `src` to `dest`, creating parent directories.
595    async fn copy_single_file(&self, src: &Path, dest: &Path, guest_dest: &str) -> Result<()> {
596        // Sanity checks: src is in app_root...
597        src.strip_prefix(&self.app_root)?;
598        // ...and dest is in the Copy root.
599        if let FilesMountStrategy::Copy(files_mount_root) = &self.files_mount_strategy {
600            dest.strip_prefix(files_mount_root)?;
601        } else {
602            unreachable!();
603        }
604
605        let _loading_permit = self.file_loading_permits.acquire().await?;
606        let dest_parent = parent_dir(dest)?;
607        crate::fs::create_dir_all(&dest_parent)
608            .await
609            .with_context(|| {
610                format!(
611                    "Failed to create parent directory {}",
612                    quoted_path(&dest_parent)
613                )
614            })?;
615        crate::fs::copy(src, dest)
616            .await
617            .or_else(|e| Self::failed_to_copy_single_file_error(src, dest, guest_dest, e))?;
618        tracing::debug!("Copied {src:?} to {dest:?}");
619        Ok(())
620    }

622    fn failed_to_copy_single_file_error<T>(
623        src: &Path,
624        dest: &Path,
625        guest_dest: &str,
626        e: anyhow::Error,
627    ) -> anyhow::Result<T> {
628        let src_text = quoted_path(src);
629        let dest_text = quoted_path(dest);
630        let base_msg = format!("Failed to copy {src_text} to working path {dest_text}");
631
632        if let Some(io_error) = e.downcast_ref::<std::io::Error>() {
633            if Self::is_directory_like(guest_dest)
634                || io_error.kind() == std::io::ErrorKind::NotFound
635            {
636                return Err(anyhow::anyhow!(
637                    r#""{guest_dest}" is not a valid destination file name"#
638                ))
639                .context(base_msg);
640            }
641        }
642
643        Err(e).with_context(|| format!("{base_msg} (for destination path \"{guest_dest}\")"))
644    }

646    /// Does a guest path appear to be a directory name, e.g. "/" or ".."? This is for guest
647    /// paths *only* and does not consider Windows separators.
648    fn is_directory_like(guest_path: &str) -> bool {
649        guest_path.ends_with('/') || guest_path.ends_with('.') || guest_path.ends_with("..")
650    }

652    // Resolve the given direct mount directory, checking that it is valid for
653    // direct mounting and returning its canonicalized source path.
654    async fn resolve_direct_mount(&self, mount: &WasiFilesMount) -> Result<ContentPath> {
655        let (src, dest) = match mount {
656            WasiFilesMount::Pattern(pattern) => (pattern, pattern),
657            WasiFilesMount::Placement {
658                source,
659                destination,
660            } => (source, destination),
661        };
662        let path = self.app_root.join(src);
663        if !path.is_dir() {
664            bail!("Only directory mounts are supported with `--direct-mounts`; {src:?} is not a directory.");
665        }
666        Ok(ContentPath {
667            content: file_content_ref(src)?,
668            path: dest.into(),
669        })
670    }

673fn explain_file_mount_source_error(e: anyhow::Error, src: &Path) -> anyhow::Error {
674    if let Some(io_error) = e.downcast_ref::<std::io::Error>() {
675        if io_error.kind() == std::io::ErrorKind::NotFound {
676            return anyhow::anyhow!("File or directory {} does not exist", quoted_path(src));
677        }
678    }
679    e.context(format!("invalid file mount source {}", quoted_path(src)))
680}

709fn locked_metadata(
710    details: v2::AppDetails,
711    trigger_types: impl Iterator<Item = String>,
712) -> Result<ValuesMap> {
713    let mut builder = ValuesMapBuilder::new();
714    builder
715        .string("name", details.name)
716        .string("version", details.version)
717        .string("description", details.description)
718        .string_array("authors", details.authors)
719        .serializable("triggers", &details.trigger_global_configs)?;
720
721    // Duplicate single-trigger global options into "trigger" with "type"
722    // key to maintain backward compatibility for a while.
723    let types = trigger_types.collect::<Vec<_>>();
724    if types.len() == 1 {
725        let trigger_type = types.into_iter().next().unwrap();
726        let mut single_trigger = details
727            .trigger_global_configs
728            .get(&trigger_type)
729            .cloned()
730            .unwrap_or_default();
731        single_trigger.insert("type".into(), trigger_type.into());
732        builder.serializable("trigger", single_trigger).unwrap();
733    }
734
735    Ok(builder.build())
736}

749fn locked_trigger(trigger_type: String, trigger: v2::Trigger) -> Result<LockedTrigger> {
750    fn reference_id(spec: v2::ComponentSpec) -> toml::Value {
751        let v2::ComponentSpec::Reference(id) = spec else {
752            unreachable!("should have already been normalized");
753        };
754        id.as_ref().into()
755    }
756
757    let mut config = trigger.config;
758    if let Some(id) = trigger.component.map(reference_id) {
759        config.insert("component".into(), id);
760    }
761    if !trigger.components.is_empty() {
762        // Flatten trigger config `components` `OneOrManyComponentSpecs` into
763        // lists of component references.
764        config.insert(
765            "components".into(),
766            trigger
767                .components
768                .into_iter()
769                .map(|(key, specs)| {
770                    (
771                        key,
772                        specs
773                            .0
774                            .into_iter()
775                            .map(reference_id)
776                            .collect::<Vec<_>>()
777                            .into(),
778                    )
779                })
780                .collect::<toml::Table>()
781                .into(),
782        );
783    }
784
785    Ok(LockedTrigger {
786        id: trigger.id,
787        trigger_type,
788        trigger_config: config.try_into()?,
789    })
790}

804fn file_url(path: impl AsRef<Path>) -> Result<String> {
805    let path = path.as_ref();
806    let abs_path = safe_canonicalize(path)
807        .with_context(|| format!("Couldn't resolve `{}`", path.display()))?;
808    Ok(Url::from_file_path(abs_path).unwrap().to_string())
809}

811fn requires_service_chaining(component: &spin_manifest::schema::v2::Component) -> bool {
812    component
813        .normalized_allowed_outbound_hosts()
814        .unwrap_or_default()
815        .iter()
816        .any(|h| is_chaining_host(h))
817}
818
819fn is_chaining_host(pattern: &str) -> bool {
820    use spin_factor_outbound_networking::{AllowedHostConfig, HostConfig};
821
822    let Ok(allowed) = AllowedHostConfig::parse(pattern) else {
823        return false;
824    };
825
826    match allowed.host() {
827        HostConfig::List(hosts) => hosts
828            .iter()
829            .any(|h| h.ends_with(SERVICE_CHAINING_DOMAIN_SUFFIX)),
830        HostConfig::AnySubdomain(domain) => domain == SERVICE_CHAINING_DOMAIN_SUFFIX,
831        _ => false,
832    }
833}
