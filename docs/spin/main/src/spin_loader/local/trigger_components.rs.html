5/// We want all component/composition graph information to be in the component,
6/// because the component ID is how Spin looks this stuff up. So if a trigger
7/// contains a `dependencies` table, e.g. specifying middleware, we want to move
8/// that to the component.
9///
10/// But it's possible to have two triggers pointing to the same primary component,
11/// but with different middleware. In this case, we will synthesise a component
12/// for each such trigger, with the same main configuration but with its own
13/// "extra" components.
14pub fn reassign_trigger_deps(mut locked: LockedApp) -> LockedApp {
15    let mut id_dispenser = SyntheticIdDispenser::new();
16
17    let needs_splitting = needs_splitting(&locked);
18
19    for (component_to_split, triggers) in needs_splitting {
20        for trigger in &triggers {
21            if !has_trigger_deps(trigger) {
22                // It's possible to have e.g. 3 triggers pointing to the same component,
23                // with only one enriched with middleware. The two unenriched ones can
24                // continue pointing to the original component.
25                continue;
26            }
27
28            // We need to split off a munge for this component-trigger combination.
29            // Locate the component, clone it under a new ID, and add the new-named clone
30            // to the app. Then point the trigger at the new name.
31            let mut split_out_component = locked
32                .components
33                .iter()
34                .find(|c| c.id == *component_to_split)
35                .unwrap()
36                .clone();
37
38            let synthetic_id = id_dispenser.create_id(&trigger.id, &component_to_split);
39            split_out_component.id = synthetic_id.clone();
40            locked.components.push(split_out_component);
41            set_component_id(&mut locked, &trigger.id, &synthetic_id);
42        }
43    }
44
45    // Now we have cloned components so that each set of { primary + trigger extras }
46    // can have its own component, meaning that composition graphs remain uniquely
47    // identified by component ID. So we can move all extra trigger components to
48    // the now-unique components, where they can later undergo trigger-specific
49    // composition.
50    move_deps_from_triggers_to_components(&mut locked);
51
52    locked
53}

55fn needs_splitting(locked: &LockedApp) -> HashMap<String, Vec<LockedTrigger>> {
56    let referenced_component_ids: Vec<_> =
57        locked.triggers.iter().filter_map(component_id).collect();
58    let cid_to_triggers: HashMap<_, _> = referenced_component_ids
59        .iter()
60        .map(|cid| (cid.clone(), triggers_referencing(&locked.triggers, cid)))
61        .collect();
62    cid_to_triggers
63        .into_iter()
64        .filter(|(_, triggers)| triggers.len() > 1 && triggers.iter().any(has_trigger_deps))
65        .collect::<HashMap<_, _>>()
66}

68fn move_deps_from_triggers_to_components(locked: &mut LockedApp) {
69    for trigger in &mut locked.triggers {
70        if has_trigger_deps(trigger)
71            && let Some(component_id) = component_id(trigger)
72            && let Some(component) = get_component_mut(&mut locked.components, &component_id)
73        {
74            component.trigger_dependencies = trigger.trigger_dependencies.clone();
75            component.metadata.insert(
76                "resolve-trigger-dependencies-using".into(),
77                trigger.trigger_type.clone().into(),
78            );
79            trigger.trigger_dependencies.clear();
80        }
81    }
82}

84fn get_component_mut<'a>(
85    components: &'a mut [LockedComponent],
86    component_id: &str,
87) -> Option<&'a mut LockedComponent> {
88    components.iter_mut().find(|c| c.id == component_id)
89}

91fn component_id(trigger: &LockedTrigger) -> Option<String> {
92    trigger
93        .trigger_config
94        .get("component")
95        .and_then(|v| v.as_str())
96        .map(|s| s.to_string())
97}
98
99fn set_component_id(app: &mut LockedApp, trigger_id: &str, component_id: &str) {
100    let trigger = app
101        .triggers
102        .iter_mut()
103        .find(|t| t.id == trigger_id)
104        .unwrap();
105    trigger
106        .trigger_config
107        .as_object_mut()
108        .unwrap()
109        .insert("component".into(), component_id.into());
110}
111
112fn has_trigger_deps(trigger: &LockedTrigger) -> bool {
113    let empty = trigger.trigger_dependencies.is_empty()
114        || trigger
115            .trigger_dependencies
116            .iter()
117            .all(|(_, tds)| tds.is_empty());
118    !empty
119}

121fn triggers_referencing(all_triggers: &[LockedTrigger], cid: &String) -> Vec<LockedTrigger> {
122    all_triggers
123        .iter()
124        .filter(|t| component_id(t).as_ref() == Some(cid))
125        .cloned()
126        .collect()
127}

129/// Helper for generating synthetic IDs for split-out components.
130/// Just keeps a bit of faffy gunk out of the main flow.
131struct SyntheticIdDispenser {
132    seen: HashSet<String>,
133    disambiguator: u32,
134}
135
136impl SyntheticIdDispenser {
137    fn new() -> Self {
138        Self {
139            seen: HashSet::new(),
140            disambiguator: 0,
141        }
142    }
143    fn create_id(&mut self, trigger_id: &str, component_id: &str) -> String {
144        let mut synthetic_id = format!("{component_id}-for-{}", trigger_id);
145        if self.seen.contains(&synthetic_id) {
146            self.disambiguator += 1;
147            synthetic_id = format!("{synthetic_id}-d{}", self.disambiguator);
148        }
149        self.seen.insert(synthetic_id.clone());
150        synthetic_id
151    }
152}
