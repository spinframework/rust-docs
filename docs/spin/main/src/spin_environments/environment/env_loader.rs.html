21/// Load all the listed environments from their registries or paths.
22/// Registry data will be cached, with a lockfile under `.spin` mapping
23/// environment IDs to digests (to allow cache lookup without needing
24/// to fetch the digest from the registry).
25pub async fn load_environments<'a>(
26    env_ids: &[&'a TargetEnvironmentRef],
27    cache_root: Option<std::path::PathBuf>,
28    app_dir: &std::path::Path,
29) -> anyhow::Result<HashMap<&'a TargetEnvironmentRef, Arc<TargetEnvironment>>> {
30    if env_ids.is_empty() {
31        return Ok(Default::default());
32    }
33
34    let cache = spin_loader::cache::Cache::new(cache_root)
35        .await
36        .context("Unable to create cache")?;
37    let lockfile_dir = app_dir.join(".spin");
38    let lockfile_path = lockfile_dir.join("target-environments.lock");
39
40    let orig_lockfile: TargetEnvironmentLockfile = tokio::fs::read_to_string(&lockfile_path)
41        .await
42        .ok()
43        .and_then(|s| serde_json::from_str(&s).ok())
44        .unwrap_or_default();
45    let lockfile = std::sync::Arc::new(tokio::sync::RwLock::new(orig_lockfile.clone()));
46
47    let envs = try_join_all(
48        env_ids
49            .iter()
50            .map(|e| load_environment(e, app_dir, &cache, &lockfile)),
51    )
52    .await?
53    .into_iter()
54    .map(|(k, v)| (k, Arc::new(v)))
55    .collect();
56
57    let final_lockfile = &*lockfile.read().await;
58    if *final_lockfile != orig_lockfile
59        && let Ok(lockfile_json) = serde_json::to_string_pretty(&final_lockfile)
60    {
61        _ = tokio::fs::create_dir_all(lockfile_dir).await;
62        _ = tokio::fs::write(&lockfile_path, lockfile_json).await; // failure to update lockfile is not an error
63    }
64
65    Ok(envs)
66}

68/// Loads the given `TargetEnvironment` from a registry or directory.
69async fn load_environment<'a>(
70    env_id: &'a TargetEnvironmentRef,
71    app_dir: &Path,
72    cache: &spin_loader::cache::Cache,
73    lockfile: &std::sync::Arc<tokio::sync::RwLock<TargetEnvironmentLockfile>>,
74) -> anyhow::Result<(&'a TargetEnvironmentRef, TargetEnvironment)> {
75    let env = match env_id {
76        TargetEnvironmentRef::Catalogue(id) => {
77            load_environment_from_catalogue(id, cache, lockfile).await
78        }
79        TargetEnvironmentRef::Http { url } => {
80            load_environment_from_http(url, cache, lockfile).await
81        }
82        TargetEnvironmentRef::File { path } => {
83            load_environment_from_file(app_dir.join(path), cache, lockfile).await
84        }
85    }?;
86    Ok((env_id, env))
87}

89/// Loads a `TargetEnvironment` from the catalogue. If not found, the catalogue is refreshed
90/// and retried. Any remote packages the environment references will be used
91/// from cache if available; otherwise, they will be saved to the cache, and the
92/// in-memory lockfile object updated.
93async fn load_environment_from_catalogue(
94    env_id: &str,
95    cache: &spin_loader::cache::Cache,
96    lockfile: &std::sync::Arc<tokio::sync::RwLock<TargetEnvironmentLockfile>>,
97) -> anyhow::Result<TargetEnvironment> {
98    let catalogue = Catalogue::try_default()?;
99    let env_id = env_id.replace(':', "@"); // back compat
100    let env_def = match catalogue.get(&env_id).await? {
101        Some(env_def) => env_def,
102        None => {
103            catalogue.update().await?;
104            catalogue
105                .get(&env_id)
106                .await?
107                .with_context(|| anyhow!("Cannot load target environment '{env_id}'"))?
108        }
109    };
110    load_environment_from_env_def(&env_id, env_def, None, cache, lockfile).await
111}

113/// Loads a `TargetEnvironment` from the environment definition at the given
114/// URL. Any remote packages the environment references will be used
115/// from cache if available; otherwise, they will be saved to the cache, and the
116/// in-memory lockfile object updated.
117async fn load_environment_from_http(
118    url: &str,
119    cache: &spin_loader::cache::Cache,
120    lockfile: &std::sync::Arc<tokio::sync::RwLock<TargetEnvironmentLockfile>>,
121) -> anyhow::Result<TargetEnvironment> {
122    let toml_text = reqwest::get(url).await?.text().await?;
123    let env_def: EnvironmentDefinition = toml::from_str(&toml_text)?;
124
125    let url = url::Url::parse(url)?;
126    let env_id = url
127        .path_segments()
128        .with_context(|| format!("environment URL {url} does not have a path"))?
129        .next_back()
130        .with_context(|| format!("environment URL {url} does not have a path"))?;
131    let env_id = env_id
132        .rsplit_once('.')
133        .map(|(stem, _)| stem)
134        .unwrap_or(env_id);
135    load_environment_from_env_def(env_id, env_def, None, cache, lockfile).await
136}

138/// Loads a `TargetEnvironment` from the given TOML file. Any remote packages
139/// it references will be used from cache if available; otherwise, they will be saved
140/// to the cache, and the in-memory lockfile object updated.
141async fn load_environment_from_file(
142    path: impl AsRef<Path>,
143    cache: &spin_loader::cache::Cache,
144    lockfile: &std::sync::Arc<tokio::sync::RwLock<TargetEnvironmentLockfile>>,
145) -> anyhow::Result<TargetEnvironment> {
146    let path = path.as_ref();
147    let env_def_dir = path.parent();
148    let name = path
149        .file_stem()
150        .and_then(|s| s.to_str())
151        .map(|s| s.to_owned())
152        .unwrap();
153    let toml_text = tokio::fs::read_to_string(path).await.with_context(|| {
154        format!(
155            "unable to read target environment from {}",
156            quoted_path(path)
157        )
158    })?;
159    let env_def: EnvironmentDefinition = toml::from_str(&toml_text)?;
160    load_environment_from_env_def(&name, env_def, env_def_dir, cache, lockfile).await
161}

163/// Loads a `TargetEnvironment` from the given TOML text. Any remote packages
164/// it references will be used from cache if available; otherwise, they will be saved
165/// to the cache, and the in-memory lockfile object updated.
166async fn load_environment_from_env_def(
167    name: &str,
168    env: EnvironmentDefinition,
169    relative_to_dir: Option<&Path>,
170    cache: &spin_loader::cache::Cache,
171    lockfile: &std::sync::Arc<tokio::sync::RwLock<TargetEnvironmentLockfile>>,
172) -> anyhow::Result<TargetEnvironment> {
173    let mut trigger_worlds = HashMap::new();
174    let mut trigger_capabilities = HashMap::new();
175
176    // TODO: parallel all the things
177    // TODO: this loads _all_ triggers not just the ones we need
178    for (trigger_type, trigger_env) in env.triggers() {
179        trigger_worlds.insert(
180            trigger_type.to_owned(),
181            load_worlds(trigger_env.world_refs(), relative_to_dir, cache, lockfile).await?,
182        );
183        trigger_capabilities.insert(trigger_type.to_owned(), trigger_env.capabilities());
184    }
185
186    let unknown_trigger = match env.default() {
187        None => UnknownTrigger::Deny,
188        Some(env) => UnknownTrigger::Allow(
189            load_worlds(env.world_refs(), relative_to_dir, cache, lockfile).await?,
190        ),
191    };
192    let unknown_capabilities = match env.default() {
193        None => vec![],
194        Some(env) => env.capabilities(),
195    };
196
197    Ok(TargetEnvironment {
198        name: name.to_owned(),
199        trigger_worlds,
200        trigger_capabilities,
201        unknown_trigger,
202        unknown_capabilities,
203    })
204}

206async fn load_worlds(
207    world_refs: &[WorldRef],
208    relative_to_dir: Option<&Path>,
209    cache: &spin_loader::cache::Cache,
210    lockfile: &std::sync::Arc<tokio::sync::RwLock<TargetEnvironmentLockfile>>,
211) -> anyhow::Result<CandidateWorlds> {
212    let mut worlds = vec![];
213
214    for world_ref in world_refs {
215        worlds.push(load_world(world_ref, relative_to_dir, cache, lockfile).await?);
216    }
217
218    Ok(CandidateWorlds { worlds })
219}

221async fn load_world(
222    world_ref: &WorldRef,
223    relative_to_dir: Option<&Path>,
224    cache: &spin_loader::cache::Cache,
225    lockfile: &std::sync::Arc<tokio::sync::RwLock<TargetEnvironmentLockfile>>,
226) -> anyhow::Result<CandidateWorld> {
227    match world_ref {
228        WorldRef::DefaultRegistry(world) => {
229            load_world_from_registry(DEFAULT_PACKAGE_REGISTRY, world, cache, lockfile).await
230        }
231        WorldRef::Registry { registry, world } => {
232            load_world_from_registry(registry, world, cache, lockfile).await
233        }
234        WorldRef::WitDirectory { path, world } => {
235            let path = match relative_to_dir {
236                Some(dir) => dir.join(path),
237                None => path.to_owned(),
238            };
239            load_world_from_dir(&path, world)
240        }
241    }
242}

244fn load_world_from_dir(
245    path: impl AsRef<Path>,
246    world: &WorldName,
247) -> anyhow::Result<CandidateWorld> {
248    let path = path.as_ref();
249    let mut resolve = wit_parser::Resolve::default();
250    let (pkg_id, _) = resolve.push_dir(path)?;
251    let decoded = wit_parser::decoding::DecodedWasm::WitPackage(resolve, pkg_id);
252    CandidateWorld::from_decoded_wasm(world, path, decoded)
253}

255/// Loads the given `TargetEnvironment` from the given registry, or
256/// from cache if available. If the environment is not in cache, the
257/// encoded WIT will be cached, and the in-memory lockfile object
258/// updated.
259async fn load_world_from_registry(
260    registry: &str,
261    world_name: &WorldName,
262    cache: &spin_loader::cache::Cache,
263    lockfile: &std::sync::Arc<tokio::sync::RwLock<TargetEnvironmentLockfile>>,
264) -> anyhow::Result<CandidateWorld> {
265    use futures_util::TryStreamExt;
266
267    if let Some(digest) = lockfile
268        .read()
269        .await
270        .package_digest(registry, world_name.package())
271        && let Ok(cache_file) = cache.wasm_file(digest)
272        && let Ok(bytes) = tokio::fs::read(&cache_file).await
273    {
274        return CandidateWorld::from_package_bytes(world_name, bytes);
275    }
276
277    let pkg_name = world_name.package_namespaced_name();
278    let pkg_ref = world_name.package_ref()?;
279
280    let wkg_registry: wasm_pkg_client::Registry = registry
281        .parse()
282        .with_context(|| format!("Registry {registry} is not a valid registry name"))?;
283
284    let mut wkg_config = wasm_pkg_client::Config::global_defaults().await?;
285    wkg_config.set_package_registry_override(
286        pkg_ref,
287        wasm_pkg_client::RegistryMapping::Registry(wkg_registry),
288    );
289
290    let client = wasm_pkg_client::Client::new(wkg_config);
291
292    let package = pkg_name
293        .to_owned()
294        .try_into()
295        .with_context(|| format!("Failed to parse environment name {pkg_name} as package name"))?;
296    let version = world_name
297        .package_version() // TODO: surely we can cope with worlds from unversioned packages? surely?
298        .ok_or_else(|| anyhow!("{world_name} is unversioned: this is not currently supported"))?;
299
300    let release = client
301        .get_release(&package, version)
302        .await
303        .with_context(|| format!("Failed to get {} from registry", world_name.package()))?;
304    let stm = client
305        .stream_content(&package, &release)
306        .await
307        .with_context(|| format!("Failed to get {} from registry", world_name.package()))?;
308    let bytes = stm
309        .try_collect::<bytes::BytesMut>()
310        .await
311        .with_context(|| format!("Failed to get {} from registry", world_name.package()))?
312        .to_vec();
313
314    let digest = release.content_digest.to_string();
315    _ = cache.write_wasm(&bytes, &digest).await; // Failure to cache is not fatal
316    lockfile
317        .write()
318        .await
319        .set_package_digest(registry, world_name.package(), &digest);
320
321    CandidateWorld::from_package_bytes(world_name, bytes)
322}
