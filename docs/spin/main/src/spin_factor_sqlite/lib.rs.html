38        ctx.link_bindings(v1::add_to_linker)?;

43    fn configure_app<T: spin_factors::RuntimeFactors>(
44        &self,
45        mut ctx: spin_factors::ConfigureAppContext<T, Self>,
46    ) -> anyhow::Result<Self::AppState> {
47        let connection_creators = ctx
48            .take_runtime_config()
49            .unwrap_or_default()
50            .connection_creators;
51
52        let allowed_databases = ctx
53            .app()
54            .components()
55            .map(|component| {
56                Ok((
57                    component.id().to_string(),
58                    Arc::new(
59                        component
60                            .get_metadata(ALLOWED_DATABASES_KEY)?
61                            .unwrap_or_default()
62                            .into_iter()
63                            .collect::<HashSet<_>>(),
64                    ),
65                ))
66            })
67            .collect::<anyhow::Result<HashMap<_, _>>>()?;
68
69        ensure_allowed_databases_are_configured(&allowed_databases, |label| {
70            connection_creators.contains_key(label)
71        })?;
72
73        Ok(AppState::new(allowed_databases, connection_creators))
74    }
75
76    fn prepare<T: spin_factors::RuntimeFactors>(
77        &self,
78        ctx: spin_factors::PrepareContext<T, Self>,
79    ) -> spin_factors::anyhow::Result<Self::InstanceBuilder> {
80        let allowed_databases = ctx
81            .app_state()
82            .allowed_databases
83            .get(ctx.app_component().id())
84            .cloned()
85            .unwrap_or_default();
86        Ok(InstanceState::new(
87            allowed_databases,
88            ctx.app_state().connection_creators.clone(),
89        ))
90    }

93/// Ensure that all the databases in the allowed databases list for each component are configured
94fn ensure_allowed_databases_are_configured(
95    allowed_databases: &HashMap<String, Arc<HashSet<String>>>,
96    is_configured: impl Fn(&str) -> bool,
97) -> anyhow::Result<()> {
98    let mut errors = Vec::new();
99    for (component_id, allowed_dbs) in allowed_databases {
100        for allowed in allowed_dbs.iter() {
101            if !is_configured(allowed) {
102                errors.push(format!(
103                    "- Component {component_id} uses database '{allowed}'"
104                ));
105            }
106        }
107    }
108
109    if !errors.is_empty() {
110        let prologue = vec![
111            "One or more components use SQLite databases which are not defined.",
112            "Check the spelling, or pass a runtime configuration file that defines these stores.",
113            "See https://spinframework.dev/dynamic-configuration#sqlite-storage-runtime-configuration",
114            "Details:",
115        ];
116        let lines: Vec<_> = prologue
117            .into_iter()
118            .map(|s| s.to_owned())
119            .chain(errors)
120            .collect();
121        return Err(anyhow::anyhow!(lines.join("\n")));
122    }
123    Ok(())
124}

129#[derive(Clone)]
130pub struct AppState {
131    /// A map from component id to a set of allowed database labels.
132    allowed_databases: HashMap<String, Arc<HashSet<String>>>,
133    /// A mapping from database label to a connection creator.
134    connection_creators: HashMap<String, Arc<dyn ConnectionCreator>>,
135}
136
137impl AppState {
138    /// Create a new `AppState`
139    pub fn new(
140        allowed_databases: HashMap<String, Arc<HashSet<String>>>,
141        connection_creators: HashMap<String, Arc<dyn ConnectionCreator>>,
142    ) -> Self {
143        Self {
144            allowed_databases,
145            connection_creators,
146        }
147    }
148
149    /// Get a connection for a given database label.
150    ///
151    /// Returns `None` if there is no connection creator for the given label.
152    pub async fn get_connection(
153        &self,
154        label: &str,
155    ) -> Option<Result<Box<dyn Connection>, v2::Error>> {
156        let connection = self
157            .connection_creators
158            .get(label)?
159            .create_connection(label)
160            .await;
161        Some(connection)
162    }
163
164    /// Returns true if the given database label is used by any component.
165    pub fn database_is_used(&self, label: &str) -> bool {
166        self.allowed_databases
167            .values()
168            .any(|stores| stores.contains(label))
169    }
170}

172/// A creator of a connections for a particular SQLite database.
173#[async_trait]
174pub trait ConnectionCreator: Send + Sync {
175    /// Get a *new* [`Connection`]
176    ///
177    /// The connection should be a new connection, not a reused one.
178    async fn create_connection(
179        &self,
180        label: &str,
181    ) -> Result<Box<dyn Connection + 'static>, v2::Error>;
182}
183
184#[async_trait]
185impl<F> ConnectionCreator for F
186where
187    F: Fn() -> anyhow::Result<Box<dyn Connection + 'static>> + Send + Sync + 'static,
188{
189    async fn create_connection(
190        &self,
191        label: &str,
192    ) -> Result<Box<dyn Connection + 'static>, v2::Error> {
193        let _ = label;
194        (self)().map_err(|_| v2::Error::InvalidConnection)
195    }
196}
