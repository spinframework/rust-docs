59impl spin::sqlite::sqlite::Value {
60    pub fn memory_size(&self) -> usize {
61        match self {
62            Self::Null | Self::Integer(_) | Self::Real(_) => std::mem::size_of::<Self>(),
63            Self::Text(t) => std::mem::size_of::<Self>() + t.len(),
64            Self::Blob(b) => std::mem::size_of::<Self>() + b.len(),
65        }
66    }
67}

69impl spin::postgres4_1_0::postgres::DbValue {
70    pub fn memory_size(&self) -> usize {
71        match self {
72            Self::DbNull
73            | Self::Boolean(_)
74            | Self::Int8(_)
75            | Self::Int16(_)
76            | Self::Int32(_)
77            | Self::Int64(_)
78            | Self::Floating32(_)
79            | Self::Floating64(_)
80            | Self::Date(_)
81            | Self::Time(_)
82            | Self::Datetime(_)
83            | Self::Timestamp(_)
84            | Self::RangeInt32(_)
85            | Self::RangeInt64(_)
86            | Self::RangeDecimal(_)
87            | Self::Interval(_) => std::mem::size_of::<Self>(),
88            Self::ArrayInt32(v) => {
89                std::mem::size_of::<Self>() + (v.len() * std::mem::size_of::<Option<i32>>())
90            }
91            Self::ArrayInt64(v) => {
92                std::mem::size_of::<Self>() + (v.len() * std::mem::size_of::<Option<i64>>())
93            }
94            Self::ArrayDecimal(v) | Self::ArrayStr(v) => {
95                std::mem::size_of::<Self>()
96                    + v.iter()
97                        .map(|v| {
98                            std::mem::size_of::<Option<String>>()
99                                + v.as_ref().map(|s| s.len()).unwrap_or(0)
100                        })
101                        .sum::<usize>()
102            }
103            Self::Str(s) | Self::Uuid(s) | Self::Decimal(s) => {
104                std::mem::size_of::<Self>() + s.len()
105            }
106            Self::Jsonb(b) | Self::Binary(b) | Self::Unsupported(b) => {
107                std::mem::size_of::<Self>() + b.len()
108            }
109        }
110    }
111}
