18/// An [`HttpExecutor`] that uses the `fermyon:spin/inbound-http` interface.
19#[derive(Clone)]
20pub struct SpinHttpExecutor;
21
22impl HttpExecutor for SpinHttpExecutor {
23    #[instrument(name = "spin_trigger_http.execute_wasm", skip_all, err(level = Level::INFO), fields(otel.name = format!("execute_wasm_component {}", route_match.lookup_key().to_string())))]
24    async fn execute<F: RuntimeFactors>(
25        &self,
26        instance_builder: TriggerInstanceBuilder<'_, F>,
27        route_match: &RouteMatch<'_, '_>,
28        req: Request<Body>,
29        client_addr: SocketAddr,
30    ) -> Result<Response<Body>> {
31        let spin_http::routes::TriggerLookupKey::Component(component_id) = route_match.lookup_key()
32        else {
33            unreachable!()
34        };
35
36        tracing::trace!("Executing request using the Spin executor for component {component_id}");
37
38        let (instance, mut store) = instance_builder.instantiate(()).await?;
39
40        let headers = prepare_request_headers(&req, route_match, client_addr)?;
41        // Expects here are safe since we have already checked that this
42        // instance exists
43        let inbound_http = instance
44            .get_export_index(&mut store, None, "fermyon:spin/inbound-http")
45            .expect("no fermyon:spin/inbound-http found");
46        let handle_request = instance
47            .get_export_index(&mut store, Some(&inbound_http), "handle-request")
48            .expect("no handle-request found");
49        let func = instance.get_typed_func::<(http_types::Request,), (http_types::Response,)>(
50            &mut store,
51            &handle_request,
52        )?;
53
54        let (parts, body) = req.into_parts();
55        let bytes = body.collect().await?.to_bytes().to_vec();
56
57        let method = if let Some(method) = convert_method(&parts.method) {
58            method
59        } else {
60            return Ok(Response::builder()
61                .status(http::StatusCode::METHOD_NOT_ALLOWED)
62                .body(body::empty())?);
63        };
64
65        // Preparing to remove the params field. We are leaving it in place for now
66        // to avoid breaking the ABI, but no longer pass or accept values in it.
67        // https://github.com/spinframework/spin/issues/663
68        let params = vec![];
69
70        let uri = match parts.uri.path_and_query() {
71            Some(u) => u.to_string(),
72            None => parts.uri.to_string(),
73        };
74
75        let req = http_types::Request {
76            method,
77            uri,
78            headers,
79            params,
80            body: Some(bytes),
81        };
82
83        let (resp,) = func.call_async(&mut store, (req,)).await?;
84
85        if resp.status < 100 || resp.status > 600 {
86            tracing::error!("malformed HTTP status code");
87            return Ok(Response::builder()
88                .status(http::StatusCode::INTERNAL_SERVER_ERROR)
89                .body(body::empty())?);
90        };
91
92        let mut response = http::Response::builder().status(resp.status);
93        if let Some(headers) = response.headers_mut() {
94            append_headers(headers, resp.headers)?;
95        }
96
97        let body = match resp.body {
98            Some(b) => body::full(b.into()),
99            None => body::empty(),
100        };
101
102        Ok(response.body(body)?)
103    }
104}

106fn convert_method(m: &http::Method) -> Option<http_types::Method> {
107    Some(match *m {
108        http::Method::GET => http_types::Method::Get,
109        http::Method::POST => http_types::Method::Post,
110        http::Method::PUT => http_types::Method::Put,
111        http::Method::DELETE => http_types::Method::Delete,
112        http::Method::PATCH => http_types::Method::Patch,
113        http::Method::HEAD => http_types::Method::Head,
114        http::Method::OPTIONS => http_types::Method::Options,
115        _ => return None,
116    })
