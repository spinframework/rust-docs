13#[spin_core::async_trait]
14impl TriggerDependenciesComposer for HttpMiddlewareComposer {
15    async fn compose_trigger_dependencies(
16        &self,
17        trigger_dependencies: &HashMap<String, Vec<TriggerDependency>>,
18        component: Vec<u8>,
19    ) -> anyhow::Result<Vec<u8>> {
20        let Some(middlewares) = trigger_dependencies.get("middleware") else {
21            return Ok(component);
22        };
23        if trigger_dependencies.len() > 1 {
24            bail!("the HTTP trigger's only allowed trigger dependency is `middleware`");
25        }
26        if middlewares.is_empty() {
27            return Ok(component);
28        }
29
30        compose_middlewares(component, middlewares).await
31    }
32}

34/// Chain a list of component packages into a middleware pipeline.
35///
36/// `packages` is ordered from **outermost** (first to receive a request) to
37/// **innermost** (the final handler).  Every component except the last must
38/// import a name equal to `import_name` and every component must export a name
39/// equal to `export_name`.  In the common middleware pattern these are the same
40/// (e.g. both `"handle"`), but they can differ if the WIT uses separate names.
41///
42/// Returns the [`NodeId`] of the alias for the outermost component's export,
43/// ready to be passed to [`CompositionGraph::export`].
44///
45/// # Errors
46///
47/// Returns an error if fewer than two packages are provided, or if any
48/// alias / argument wiring step fails.
49fn chain(
50    graph: &mut CompositionGraph,
51    packages: &[PackageId],
52    import_name: &str,
53    export_name: &str,
54) -> anyhow::Result<wac_graph::NodeId> {
55    if packages.len() < 2 {
56        bail!("chain requires at least 2 packages, got {}", packages.len());
57    }
58
59    // Start from the innermost component (last in the list) and work outward.
60    // The innermost component is instantiated first with no wiring — its
61    // unsatisfied imports (if any) will become implicit imports of the
62    // composed component.
63    let mut iter = packages.iter().rev();
64    let innermost = *iter.next().unwrap();
65    let mut instance = graph.instantiate(innermost);
66    let mut upstream_handle = graph.alias_instance_export(instance, export_name)?;
67
68    // For each remaining component (moving outward), instantiate it and
69    // wire the previous component's export into its import.
70    for &pkg in iter {
71        instance = graph.instantiate(pkg);
72        graph.set_instantiation_argument(instance, import_name, upstream_handle)?;
73        upstream_handle = graph.alias_instance_export(instance, export_name)?;
74    }
75
76    Ok(upstream_handle)
77}

79async fn compose_middlewares(
80    primary: Vec<u8>,
81    middleware_blobs: &[TriggerDependency],
82) -> anyhow::Result<Vec<u8>> {
83    use spin_compose::DependencyLike;
84
85    const MW_HANDLER_INTERFACE: &str = "wasi:http/handler@0.3.0-rc-2026-03-15";
86
87    let mut graph = CompositionGraph::new();
88    let mut package_ids: Vec<PackageId> = Vec::new();
89
90    // Register middleware packages (outermost → innermost order).
91    for (index, dep) in middleware_blobs.iter().enumerate() {
92        let bytes: Vec<u8> = match &dep.data {
93            TriggerDependencyData::InMemory(data) => data.clone(),
94            TriggerDependencyData::OnDisk(path) => tokio::fs::read(path)
95                .await
96                .with_context(|| format!("reading middleware from {}", path.display()))?,
97        };
98        let bytes = spin_componentize::componentize_if_necessary(&bytes)
99            .context("failed to componentize")?;
100        let bytes = spin_capabilities::apply_deny_adapter(&bytes, dep.dependency.inherit())?;
101        let name = format!("middleware{index}");
102        let package = Package::from_bytes(&name, None, bytes, graph.types_mut())
103            .context("parsing middleware component")?;
104        package_ids.push(graph.register_package(package)?);
105    }
106
107    // Register the primary component (innermost in the chain).
108    let package = Package::from_bytes("primary", None, primary, graph.types_mut())
109        .context("parsing primary component")?;
110    package_ids.push(graph.register_package(package)?);
111
112    // Wire the pipeline: outermost middleware → … → primary.
113    let outermost_export = chain(
114        &mut graph,
115        &package_ids,
116        MW_HANDLER_INTERFACE,
117        MW_HANDLER_INTERFACE,
118    )?;
119
120    // Export the outermost handler as the composed component's export.
121    graph.export(outermost_export, MW_HANDLER_INTERFACE)?;
122
123    Ok(graph.encode(Default::default())?)
124}
