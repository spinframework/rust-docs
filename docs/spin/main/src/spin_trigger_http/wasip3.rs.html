24/// An [`HttpExecutor`] that uses the `wasi:http@0.3.*/handler` interface.

27impl HttpExecutor for Wasip3HttpExecutor<'_> {
28    #[instrument(name = "spin_trigger_http.execute_wasm", skip_all, err(level = Level::INFO), fields(otel.name = format!("execute_wasm_component {}", route_match.lookup_key().to_string())))]
29    async fn execute<F: RuntimeFactors>(
30        &self,
31        instance_builder: TriggerInstanceBuilder<'_, F>,
32        route_match: &RouteMatch<'_, '_>,
33        mut req: http::Request<Body>,
34        client_addr: SocketAddr,
35    ) -> Result<http::Response<Body>> {
36        super::wasi::prepare_request(route_match, &mut req, client_addr)?;
37
38        let (instance, mut store) = instance_builder.instantiate(()).await?;
39
40        let getter = (|data| wasi_http::<F>(data).unwrap())
41            as fn(&mut InstanceState<F::InstanceState, ()>) -> WasiHttpCtxView<'_>;
42
43        let (request, body) = req.into_parts();
44        let body = body.map_err(spin_factor_outbound_http::p2_to_p3_error_code);
45        let request = http::Request::from_parts(request, body);
46        let (request, request_io_result) = types::Request::from_http(request);
47        let request = wasi_http::<F>(store.data_mut())?.table.push(request)?;
48
49        let guest = self.0.load(&mut store, &instance)?;
50
51        let (tx, rx) = oneshot::channel();
52        task::spawn(
53            async move {
54                instance
55                    .run_concurrent(&mut store, async move |store| {
56                        let (response, task) = guest
57                            .wasi_http_handler()
58                            .call_handle(store, request)
59                            .await?;
60                        let response = store.with(|mut store| {
61                            anyhow::Ok(wasi_http::<F>(store.get())?.table.delete(response?)?)
62                        })?;
63                        let response = store.with(|mut store| {
64                            response.into_http_with_getter(&mut store, request_io_result, getter)
65                        })?;
66
67                        let (response_tx, response_rx) = oneshot::channel::<()>();
68                        _ = tx.send(response.map(|body| BodyWithAttachment {
69                            body,
70                            _attachment: response_tx,
71                        }));
72
73                        task.block(store).await;
74
75                        // TODO: This is a temporary workaround for
76                        // https://github.com/bytecodealliance/wasmtime/issues/11703.
77                        // Remove this (and `BodyWithAttachment`) once that
78                        // issue is addressed:
79                        _ = response_rx.await;
80
81                        anyhow::Ok(())
82                    })
83                    .await?
84            }
85            .in_current_span()
86            .inspect(|result| {
87                if let Err(error) = result {
88                    // TODO: Remove this check once we've updated to Wasmtime
89                    // 38+.
90                    //
91                    // Wasmtime 37's implementation of
92                    // `Instance::run_concurrent` returns `Trap::AsyncDeadlock`
93                    // if the `AsyncFnOnce` it was given does not resolve by the
94                    // time all outstanding tasks have finished.  In this case,
95                    // it's harmless and we can ignore it.  See
96                    // https://github.com/bytecodealliance/wasmtime/pull/11756
97                    // for details.
98                    if let Some(Trap::AsyncDeadlock) = error.downcast_ref::<Trap>() {
99                        // ignore
100                    } else {
101                        tracing::error!("Component error handling request: {error:?}");
102                    }
103                }
104            }),
105        );
106
107        Ok(rx.await?.map(|body| {
108            body.map_err(spin_factor_outbound_http::p3_to_p2_error_code)
109                .boxed()
110        }))
111    }
112}
