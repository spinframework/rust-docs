26impl HttpExecutor for Wasip3HttpExecutor<'_> {
27    #[instrument(name = "spin_trigger_http.execute_wasm", skip_all, err(level = Level::INFO), fields(otel.name = format!("execute_wasm_component {}", route_match.lookup_key().to_string())))]
28    async fn execute<F: RuntimeFactors>(
29        &self,
30        instance_builder: TriggerInstanceBuilder<'_, F>,
31        route_match: &RouteMatch<'_, '_>,
32        mut req: http::Request<Body>,
33        client_addr: SocketAddr,
34    ) -> Result<http::Response<Body>> {
35        super::wasi::prepare_request(route_match, &mut req, client_addr)?;
36
37        let (instance, mut store) = instance_builder.instantiate(()).await?;
38
39        let getter = (|data| wasi_http::<F>(data).unwrap())
40            as fn(&mut InstanceState<F::InstanceState, ()>) -> WasiHttpCtxView<'_>;
41
42        let (request, body) = req.into_parts();
43        let body = body.map_err(spin_factor_outbound_http::p2_to_p3_error_code);
44        let request = http::Request::from_parts(request, body);
45        let (request, request_io_result) = types::Request::from_http(request);
46        let request = wasi_http::<F>(store.data_mut())?.table.push(request)?;
47
48        let guest = self.0.load(&mut store, &instance)?;
49
50        let (tx, rx) = oneshot::channel();
51        task::spawn(
52            async move {
53                instance
54                    .run_concurrent(&mut store, async move |store| {
55                        let (response, task) = guest
56                            .wasi_http_handler()
57                            .call_handle(store, request)
58                            .await?;
59                        let response = store.with(|mut store| {
60                            anyhow::Ok(wasi_http::<F>(store.get())?.table.delete(response?)?)
61                        })?;
62                        let response = store.with(|mut store| {
63                            response.into_http_with_getter(&mut store, request_io_result, getter)
64                        })?;
65
66                        let (response_tx, response_rx) = oneshot::channel::<()>();
67                        _ = tx.send(response.map(|body| BodyWithAttachment {
68                            body,
69                            _attachment: response_tx,
70                        }));
71
72                        task.block(store).await;
73
74                        // TODO: This is a temporary workaround for
75                        // https://github.com/bytecodealliance/wasmtime/issues/11703.
76                        // Remove this (and `BodyWithAttachment`) once that
77                        // issue is addressed:
78                        _ = response_rx.await;
79
80                        anyhow::Ok(())
81                    })
82                    .await?
83            }
84            .in_current_span()
85            .inspect(|result| {
86                if let Err(error) = result {
87                    tracing::error!("Component error handling request: {error:?}");
88                }
89            }),
90        );
91
92        Ok(rx.await?.map(|body| {
93            body.map_err(spin_factor_outbound_http::p3_to_p2_error_code)
94                .boxed()
95        }))
96    }
97}
