63    trigger_app: TriggerApp<F>,

66    // Component ID -> handler type
67    component_handler_types: HashMap<String, HandlerType>,

71    /// Create a new [`HttpServer`].
72    pub fn new(
73        listen_addr: SocketAddr,
74        tls_config: Option<TlsConfig>,
75        find_free_port: bool,
76        trigger_app: TriggerApp<F>,
77    ) -> anyhow::Result<Self> {
78        // This needs to be a vec before building the router to handle duplicate routes
79        let component_trigger_configs = Vec::from_iter(
80            trigger_app
81                .app()
82                .trigger_configs::<HttpTriggerConfig>("http")?
83                .into_iter()
84                .map(|(_, config)| (config.component.clone(), config)),
85        );
86
87        // Build router
88        let component_routes = component_trigger_configs
89            .iter()
90            .map(|(component_id, config)| (component_id.as_str(), &config.route));
91        let mut duplicate_routes = Vec::new();
92        let router = Router::build("/", component_routes, Some(&mut duplicate_routes))?;
93        if !duplicate_routes.is_empty() {
94            tracing::error!(
95                "The following component routes are duplicates and will never be used:"
96            );
97            for dup in &duplicate_routes {
98                tracing::error!(
99                    "  {}: {} (duplicate of {})",
100                    dup.replaced_id,
101                    dup.route(),
102                    dup.effective_id,
103                );
104            }
105        }
106        if router.contains_reserved_route() {
107            tracing::error!(
108                "Routes under {} are handled by the Spin runtime and will never be reached",
109                spin_http::WELL_KNOWN_PREFIX
110            );
111        }
112        tracing::trace!(
113            "Constructed router: {:?}",
114            router.routes().collect::<Vec<_>>()
115        );
116
117        // Now that router is built we can merge duplicate routes by component
118        let component_trigger_configs = HashMap::from_iter(component_trigger_configs);
119
120        let component_handler_types = component_trigger_configs
121            .iter()
122            .map(|(component_id, trigger_config)| {
123                let pre = trigger_app.get_instance_pre(component_id)?;
124                let handler_type = match &trigger_config.executor {
125                    None | Some(HttpExecutorType::Http) => {
126                        HandlerType::from_instance_pre(pre)?
127                    }
128                    Some(HttpExecutorType::Wagi(wagi_config)) => {
129                        anyhow::ensure!(
130                            wagi_config.entrypoint == "_start",
131                            "Wagi component '{component_id}' cannot use deprecated 'entrypoint' field"
132                        );
133                        HandlerType::Wagi(CommandIndices::new(pre)
134                            .context("failed to find wasi command interface for wagi executor")?)
135                    }
136                };
137                Ok((component_id.clone(), handler_type))
138            })
139            .collect::<anyhow::Result<_>>()?;
140        Ok(Self {
141            listen_addr,
142            tls_config,
143            find_free_port,
144            router,
145            trigger_app,
146            component_trigger_configs,
147            component_handler_types,
148        })
149    }

151    /// Serve incoming requests over the provided [`TcpListener`].
152    pub async fn serve(self: Arc<Self>) -> anyhow::Result<()> {
153        let listener: TcpListener = if self.find_free_port {
154            self.search_for_free_port().await?
155        } else {
156            TcpListener::bind(self.listen_addr).await.map_err(|err| {
157                if err.kind() == ErrorKind::AddrInUse {
158                    anyhow::anyhow!("{} is already in use. To have Spin search for a free port, use the --find-free-port option.", self.listen_addr)
159                } else {
160                    anyhow::anyhow!("Unable to listen on {}: {err:?}", self.listen_addr)
161                }
162            })?
163        };
164
165        if let Some(tls_config) = self.tls_config.clone() {
166            self.serve_https(listener, tls_config).await?;
167        } else {
168            self.serve_http(listener).await?;
169        }
170        Ok(())
171    }

173    async fn search_for_free_port(&self) -> anyhow::Result<TcpListener> {
174        let mut found_listener = None;
175        let mut addr = self.listen_addr;
176
177        for _ in 1..=MAX_RETRIES {
178            if addr.port() == u16::MAX {
179                anyhow::bail!(
180                    "Couldn't find a free port as we've reached the maximum port number. Consider retrying with a lower base port."
181                );
182            }
183
184            match TcpListener::bind(addr).await {
185                Ok(listener) => {
186                    found_listener = Some(listener);
187                    break;
188                }
189                Err(err) if err.kind() == ErrorKind::AddrInUse => {
190                    addr.set_port(addr.port() + 1);
191                    continue;
192                }
193                Err(err) => anyhow::bail!("Unable to listen on {addr}: {err:?}",),
194            }
195        }
196
197        found_listener.ok_or_else(|| anyhow::anyhow!(
198            "Couldn't find a free port in the range {}-{}. Consider retrying with a different base port.",
199            self.listen_addr.port(),
200            self.listen_addr.port() + MAX_RETRIES
201        ))
202    }

204    async fn serve_http(self: Arc<Self>, listener: TcpListener) -> anyhow::Result<()> {
205        self.print_startup_msgs("http", &listener)?;
206        loop {
207            let (stream, client_addr) = listener.accept().await?;
208            self.clone()
209                .serve_connection(stream, Scheme::HTTP, client_addr);
210        }
211    }
212
213    async fn serve_https(
214        self: Arc<Self>,
215        listener: TcpListener,
216        tls_config: TlsConfig,
217    ) -> anyhow::Result<()> {
218        self.print_startup_msgs("https", &listener)?;
219        let acceptor = tls_config.server_config()?;
220        loop {
221            let (stream, client_addr) = listener.accept().await?;
222            match acceptor.accept(stream).await {
223                Ok(stream) => self
224                    .clone()
225                    .serve_connection(stream, Scheme::HTTPS, client_addr),
226                Err(err) => tracing::error!(?err, "Failed to start TLS session"),
227            }
228        }
229    }

231    /// Handles incoming requests using an HTTP executor.
232    ///
233    /// This method handles well known paths and routes requests to the handler when the router
234    /// matches the requests path.
235    pub async fn handle(
236        self: &Arc<Self>,
237        mut req: Request<Body>,
238        server_scheme: Scheme,
239        client_addr: SocketAddr,
240    ) -> anyhow::Result<Response<Body>> {
241        strip_forbidden_headers(&mut req);
242
243        spin_telemetry::extract_trace_context(&req);
244
245        let path = req.uri().path().to_string();
246
247        tracing::info!("Processing request on path '{path}'");
248
249        // Handle well-known spin paths
250        if let Some(well_known) = path.strip_prefix(spin_http::WELL_KNOWN_PREFIX) {
251            return match well_known {
252                "health" => Ok(MatchedRoute::with_response_extension(
253                    Response::new(body::full(Bytes::from_static(b"OK"))),
254                    path,
255                )),
256                "info" => self.app_info(path),
257                _ => Self::not_found(NotFoundRouteKind::WellKnown),
258            };
259        }
260
261        match self.router.route(&path) {
262            Ok(route_match) => {
263                self.handle_trigger_route(req, route_match, server_scheme, client_addr)
264                    .await
265            }
266            Err(_) => Self::not_found(NotFoundRouteKind::Normal(path.to_string())),
267        }
268    }

270    /// Handles a successful route match.
271    pub async fn handle_trigger_route(
272        self: &Arc<Self>,
273        mut req: Request<Body>,
274        route_match: RouteMatch<'_, '_>,
275        server_scheme: Scheme,
276        client_addr: SocketAddr,
277    ) -> anyhow::Result<Response<Body>> {
278        set_req_uri(&mut req, server_scheme.clone())?;
279        let app_id = self
280            .trigger_app
281            .app()
282            .get_metadata(APP_NAME_KEY)?
283            .unwrap_or_else(|| "<unnamed>".into());
284
285        let component_id = route_match.component_id();
286
287        spin_telemetry::metrics::monotonic_counter!(
288            spin.request_count = 1,
289            trigger_type = "http",
290            app_id = app_id,
291            component_id = component_id
292        );
293
294        let mut instance_builder = self.trigger_app.prepare(component_id)?;
295
296        // Set up outbound HTTP request origin and service chaining
297        // The outbound HTTP factor is required since both inbound and outbound wasi HTTP
298        // implementations assume they use the same underlying wasmtime resource storage.
299        // Eventually, we may be able to factor this out to a separate factor.
300        let outbound_http = instance_builder
301            .factor_builder::<OutboundHttpFactor>()
302            .context(
303            "The wasi HTTP trigger was configured without the required wasi outbound http support",
304        )?;
305        let origin = SelfRequestOrigin::create(server_scheme, &self.listen_addr.to_string())?;
306        outbound_http.set_self_request_origin(origin);
307        outbound_http.set_request_interceptor(OutboundHttpInterceptor::new(self.clone()))?;
308
309        // Prepare HTTP executor
310        let trigger_config = self.component_trigger_configs.get(component_id).unwrap();
311        let handler_type = self.component_handler_types.get(component_id).unwrap();
312        let executor = trigger_config
313            .executor
314            .as_ref()
315            .unwrap_or(&HttpExecutorType::Http);
316
317        let res = match executor {
318            HttpExecutorType::Http => match handler_type {
319                HandlerType::Spin => {
320                    SpinHttpExecutor
321                        .execute(instance_builder, &route_match, req, client_addr)
322                        .await
323                }
324                HandlerType::Wasi0_2(_)
325                | HandlerType::Wasi2023_11_10(_)
326                | HandlerType::Wasi2023_10_18(_) => {
327                    WasiHttpExecutor { handler_type }
328                        .execute(instance_builder, &route_match, req, client_addr)
329                        .await
330                }
331                HandlerType::Wagi(_) => unreachable!(),
332            },
333            HttpExecutorType::Wagi(wagi_config) => {
334                let indices = match handler_type {
335                    HandlerType::Wagi(indices) => indices,
336                    _ => unreachable!(),
337                };
338                let executor = WagiHttpExecutor {
339                    wagi_config,
340                    indices,
341                };
342                executor
343                    .execute(instance_builder, &route_match, req, client_addr)
344                    .await
345            }
346        };
347        match res {
348            Ok(res) => Ok(MatchedRoute::with_response_extension(
349                res,
350                route_match.raw_route(),
351            )),
352            Err(err) => {
353                tracing::error!("Error processing request: {err:?}");
354                instrument_error(&err);
355                Self::internal_error(None, route_match.raw_route())
356            }
357        }
358    }

360    /// Returns spin status information.
361    fn app_info(&self, route: String) -> anyhow::Result<Response<Body>> {
362        let info = AppInfo::new(self.trigger_app.app());
363        let body = serde_json::to_vec_pretty(&info)?;
364        Ok(MatchedRoute::with_response_extension(
365            Response::builder()
366                .header("content-type", "application/json")
367                .body(body::full(body.into()))?,
368            route,
369        ))

372    /// Creates an HTTP 500 response.
373    fn internal_error(
374        body: Option<&str>,
375        route: impl Into<String>,
376    ) -> anyhow::Result<Response<Body>> {
377        let body = match body {
378            Some(body) => body::full(Bytes::copy_from_slice(body.as_bytes())),
379            None => body::empty(),
380        };
381
382        Ok(MatchedRoute::with_response_extension(
383            Response::builder()
384                .status(StatusCode::INTERNAL_SERVER_ERROR)
385                .body(body)?,
386            route,
387        ))
388    }

390    /// Creates an HTTP 404 response.
391    fn not_found(kind: NotFoundRouteKind) -> anyhow::Result<Response<Body>> {
392        use std::sync::atomic::{AtomicBool, Ordering};
393        static SHOWN_GENERIC_404_WARNING: AtomicBool = AtomicBool::new(false);
394        if let NotFoundRouteKind::Normal(route) = kind {
395            if !SHOWN_GENERIC_404_WARNING.fetch_or(true, Ordering::Relaxed)
396                && std::io::stderr().is_terminal()
397            {
398                terminal::warn!("Request to {route} matched no pattern, and received a generic 404 response. To serve a more informative 404 page, add a catch-all (/...) route.");
399            }
400        }
401        Ok(Response::builder()
402            .status(StatusCode::NOT_FOUND)
403            .body(body::empty())?)
404    }

406    fn serve_connection<S: AsyncRead + AsyncWrite + Unpin + Send + 'static>(
407        self: Arc<Self>,
408        stream: S,
409        server_scheme: Scheme,
410        client_addr: SocketAddr,
411    ) {
412        task::spawn(async move {
413            if let Err(err) = http1::Builder::new()
414                .keep_alive(true)
415                .serve_connection(
416                    TokioIo::new(stream),
417                    service_fn(move |request| {
418                        self.clone().instrumented_service_fn(
419                            server_scheme.clone(),
420                            client_addr,
421                            request,
422                        )
423                    }),
424                )
425                .await
426            {
427                tracing::warn!("Error serving HTTP connection: {err:?}");
428            }
429        });
430    }

432    async fn instrumented_service_fn(
433        self: Arc<Self>,
434        server_scheme: Scheme,
435        client_addr: SocketAddr,
436        request: Request<Incoming>,
437    ) -> anyhow::Result<Response<HyperOutgoingBody>> {
438        let span = http_span!(request, client_addr);
439        let method = request.method().to_string();
440        async {
441            let result = self
442                .handle(
443                    request.map(|body: Incoming| {
444                        body.map_err(wasmtime_wasi_http::hyper_response_error)
445                            .boxed()
446                    }),
447                    server_scheme,
448                    client_addr,
449                )
450                .await;
451            finalize_http_span(result, method)
452        }
453        .instrument(span)
454        .await
455    }

457    fn print_startup_msgs(&self, scheme: &str, listener: &TcpListener) -> anyhow::Result<()> {
458        let local_addr = listener.local_addr()?;
459        let base_url = format!("{scheme}://{local_addr:?}");
460        terminal::step!("\nServing", "{base_url}");
461        tracing::info!("Serving {base_url}");
462
463        println!("Available Routes:");
464        for (route, component_id) in self.router.routes() {
465            println!("  {component_id}: {base_url}{route}");
466            if let Some(component) = self.trigger_app.app().get_component(component_id) {
467                if let Some(description) = component.get_metadata(APP_DESCRIPTION_KEY)? {
468                    println!("    {description}");
469                }
470            }
471        }
472        Ok(())
473    }

476/// The incoming request's scheme and authority
477///
478/// The incoming request's URI is relative to the server, so we need to set the scheme and authority.
479/// Either the `Host` header or the request's URI's authority is used as the source of truth for the authority.
480/// This function will error if the authority cannot be unambiguously determined.
481fn set_req_uri(req: &mut Request<Body>, scheme: Scheme) -> anyhow::Result<()> {
482    let uri = req.uri().clone();
483    let mut parts = uri.into_parts();
484    let headers = req.headers();
485    let header_authority = headers
486        .get(http::header::HOST)
487        .map(|h| -> anyhow::Result<Authority> {
488            let host_header = h.to_str().context("'Host' header is not valid UTF-8")?;
489            host_header
490                .parse()
491                .context("'Host' header contains an invalid authority")
492        })
493        .transpose()?;
494    let uri_authority = parts.authority;
495
496    // Get authority either from request URI or from 'Host' header
497    let authority = match (header_authority, uri_authority) {
498        (None, None) => bail!("no 'Host' header present in request"),
499        (None, Some(a)) => a,
500        (Some(a), None) => a,
501        (Some(a1), Some(a2)) => {
502            // Ensure that if `req.authority` is set, it matches what was in the `Host` header
503            // https://github.com/hyperium/hyper/issues/1612
504            if a1 != a2 {
505                return Err(anyhow::anyhow!(
506                    "authority in 'Host' header does not match authority in URI"
507                ));
508            }
509            a1
510        }
511    };
512    parts.scheme = Some(scheme);
513    parts.authority = Some(authority);
514    *req.uri_mut() = Uri::from_parts(parts).unwrap();
515    Ok(())
516}
